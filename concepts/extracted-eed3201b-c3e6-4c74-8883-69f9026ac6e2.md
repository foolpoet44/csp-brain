class CompetencyAssessmentAgent:
    def __init__(self):
        self.obsidian = ObsidianMCP()  # 온톨로지 저장
        self.notion = NotionMCP()      # 평가 데이터
        self.exa = ExaSearch()         # 학습 자료 검색
        self.gmail = GmailMCP()        # 알림
    
    def assess_and_recommend(self, employee_id):
        # 1. 온톨로지에서 역량 프레임워크 가져오기
        competency_map = self.obsidian.read_note(
            "스마트팩토리_역량_온톨로지.md"
        )
        
        # 2. Notion에서 직원 평가 데이터 조회
        assessment = self.notion.query_database(
            database_id="역량평가DB",
            filter={"직원ID": employee_id}
        )
        
        # 3. 부족한 역량 식별
        gap_analysis = self.analyze_gaps(
            competency_map, 
            assessment
        )
        
        # 4. 학습 자료 검색 (Web Search)
        learning_resources = self.exa.search(
            f"스마트팩토리 {gap_analysis['약점']} 교육"
        )
        
        # 5. 개인화된 학습 경로 생성
        learning_path = self.notion.create_page(
            parent="학습경로DB",
            properties={
                "직원": employee_id,
                "약점역량": gap_analysis['약점'],
                "추천자료": learning_resources,
                "완료기한": "+3months"

        # 6. 직원에게 알림
        self.gmail.send_email(
            to=employee_email,

            body=self.generate_email_body(learning_path)
        )

    """

        # Awesome list에서 선별한 도구들

        self.supabase = SupabaseMCP()  # 인증/사용자
        self.exa = ExaSearch()         # 지식 검색
        self.gmail = GmailMCP()        # 소통
        self.browser = PlaywrightMCP() # 자동화
        
    def run_evaluation_cycle(self):
        """
        평가 사이클 자동 실행
        """
        # 1. 평가 대상자 조회
        targets = self.supabase.query(
            "SELECT * FROM employees WHERE next_eval < NOW()"
        )
        
        # 2. 각 대상자별 처리
        for employee in targets:
            # 2-1. 과거 평가 이력 분석
            history = self.notion.query_database(
                filter={"employee_id": employee.id}
            )
            
            # 2-2. 최신 리더십 트렌드 검색
            trends = self.exa.search(
                "2026 leadership evaluation best practices"
            )
            
            # 2-3. AI 피드백 생성 (Claude 활용)
            feedback = self.generate_feedback(
                history, 
                trends
            )
            
            # 2-4. 리포트 생성 및 발송

            self.gmail.send(employee.email, report)
