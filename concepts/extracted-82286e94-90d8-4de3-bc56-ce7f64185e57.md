class GoalCycle:
    """목표들의 시간적/조직적 컨테이너"""
    name: str
    start_date: date
    end_date: date
    visibility: VisibilityScope
    goals: List[Goal]
    
    def is_active(self) -> bool:
        return self.start_date <= today() <= self.end_date
    
    def progress_summary(self) -> Dict:
        """사이클 내 전체 목표의 진척도 요약"""
        return {
            'total_goals': len(self.goals),
            'completed': sum(1 for g in self.goals if g.is_completed()),
            'on_track': sum(1 for g in self.goals if g.is_on_track()),
            'at_risk': sum(1 for g in self.goals if g.is_at_risk())
        }
```

class GoalMetrics:
    """목표 관련 자동 계산 지표"""
    
    @property
    def completion_rate(self) -> float:
        """완료율"""
        return sum(kr.progress for kr in self.key_results) / len(self.key_results)
    
    @property
    def check_in_frequency(self) -> float:
        """체크인 빈도 (주당)"""
        days = (today() - self.start_date).days
        weeks = days / 7
        return len(self.check_ins) / weeks if weeks > 0 else 0
    
    @property
    def at_risk_score(self) -> float:
        """위험도 점수 (0~1)"""
        # 진척도가 예상보다 낮으면 높은 점수
        expected_progress = self.expected_progress_by_now()
        actual_progress = self.completion_rate
        gap = expected_progress - actual_progress
        return max(0, min(1, gap / 0.5))
```
