# 분석 스크립트 예시

import pandas as pd
from scipy.stats import cronbach_alpha
import numpy as np

# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
# Step 1: 데이터 로드 및 역문항 처리
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

df = pd.read_csv('survey_data.csv')

# 역문항 변환
reverse_items = ['PE3_R', 'OS5_R', 'BI4_R', 'GM3_R', 'GM4_R', 
                 'CL1_R', 'CL2_R']

for item in reverse_items:
    df[item] = 6 - df[item]  # 5점 척도이므로 6에서 뺌

# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
# Step 2: Factor별 Cronbach's Alpha
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

factors = {
    'PU': ['PU1', 'PU2', 'PU3', 'PU4'],
    'PE': ['PE1', 'PE2', 'PE3_R', 'PE4'],
    'SE': ['SE1', 'SE2', 'SE3', 'SE4'],
    'OS': ['OS1', 'OS2', 'OS3', 'OS4', 'OS5_R'],
    'BI': ['BI1', 'BI2', 'BI3', 'BI4_R'],
    'SN': ['SN1', 'SN2', 'SN3'],
    'PBC': ['PBC1', 'PBC2', 'PBC3'],
    'GM': ['GM1', 'GM2', 'GM3_R', 'GM4_R'],
    'CL': ['CL1_R', 'CL2_R', 'CL3']
}

reliability_results = {}

for factor_name, items in factors.items():
    # 해당 Factor의 문항들만 선택
    factor_data = df[items].dropna()
    
    # Cronbach's Alpha 계산
    alpha = cronbach_alpha(factor_data.values)
    
    # Item-Total Correlation (각 문항 제거 시 alpha)
    item_deleted_alphas = {}
    for item in items:
        remaining = [i for i in items if i != item]
        alpha_if_deleted = cronbach_alpha(
            factor_data[remaining].values
        )
        item_deleted_alphas[item] = alpha_if_deleted
    
    reliability_results[factor_name] = {
        'alpha': alpha,
        'n_items': len(items),
        'n_valid': len(factor_data),
        'item_deleted': item_deleted_alphas
    }

# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
# Step 3: 결과 해석 및 문항 제거 결정
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

interpretation = """
신뢰도 판정 기준:
- α ≥ 0.90: 우수
- 0.80 ≤ α < 0.90: 양호
- 0.70 ≤ α < 0.80: 수용 가능
- α < 0.70: 재검토 필요

조치:
- α < 0.70인 Factor: 문항 제거 검토
  → item_deleted_alpha가 더 높은 문항 제거
- α ≥ 0.70: 그대로 유지
"""

for factor, results in reliability_results.items():
    print(f"\n{'='*50}")
    print(f"Factor: {factor}")
    print(f"Cronbach's Alpha: {results['alpha']:.3f}")
    print(f"문항 수: {results['n_items']}")
    print(f"유효 응답: {results['n_valid']}")
    
    if results['alpha'] < 0.70:
        print("⚠️  신뢰도 낮음 - 문항 제거 검토 필요")
        print("\n각 문항 제거 시 Alpha:")
        for item, alpha in results['item_deleted'].items():
            improvement = alpha - results['alpha']
            print(f"  {item}: {alpha:.3f} "
                  f"({'+' if improvement > 0 else ''}"
                  f"{improvement:.3f})")
    elif results['alpha'] >= 0.80:
        print("✅ 신뢰도 양호")
    else:
        print("✓  신뢰도 수용 가능")

# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

for factor_name, items in factors.items():
    df[f'{factor_name}_SCORE'] = df[items].mean(axis=1)

# AVE (Average Variance Extracted) 계산
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

def calculate_ave_cr(loadings_dict):
    """
    AVE: 평균분산추출
    CR: 개념신뢰도
    """
    results = {}
    
    for factor, loadings in loadings_dict.items():
        # 요인부하량 제곱의 합
        sum_sq_loadings = sum([l**2 for l in loadings])
        
        # 오차분산의 합
        sum_error_var = sum([1 - l**2 for l in loadings])
        
        # AVE = Σ(λ²) / [Σ(λ²) + Σ(δ)]
        ave = sum_sq_loadings / (sum_sq_loadings + sum_error_var)
        
        # CR = [Σ(λ)]² / {[Σ(λ)]² + Σ(δ)}
        sum_loadings = sum(loadings)
        cr = (sum_loadings ** 2) / (
            (sum_loadings ** 2) + sum_error_var
        )
        
        results[factor] = {'AVE': ave, 'CR': cr}
    
    return results

# CFA 결과에서 요인부하량 추출 (예시)
