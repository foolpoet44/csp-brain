from rdflib import Graph, Namespace, Literal
from rdflib.namespace import SKOS, RDF

# 그래프 생성
g = Graph()
SF = Namespace("http://smartfactory.skills/")
g.bind("sf", SF)
g.bind("skos", SKOS)

# 개념 추가
g.add((SF.dataAnalysis, RDF.type, SKOS.Concept))

# 저장
g.serialize("skills.ttl", format="turtle")
