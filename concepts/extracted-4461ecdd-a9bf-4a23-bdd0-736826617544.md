```python
"""
Notion API Client Wrapper
간단하고 재사용 가능한 Notion API 래퍼
"""

from notion_client import Client
from typing import Dict, List, Optional
import os
from dotenv import load_dotenv

# 환경변수 로드
load_dotenv()


class NotionClient:
    """
    Notion API를 쉽게 사용하기 위한 래퍼 클래스
    
    철학:
    - 복잡한 Notion API를 단순화
    - 에러 핸들링 자동화
    - 재사용 가능한 메서드 제공
    """
    
    def __init__(self):
        """Notion Client 초기화"""
        self.token = os.getenv("NOTION_TOKEN")
        if not self.token:
            raise ValueError("NOTION_TOKEN이 .env에 설정되지 않았습니다.")
        
        self.client = Client(auth=self.token)
        self.parent_page_id = os.getenv("PARENT_PAGE_ID")
    
    def create_database(
        self, 
        title: str, 
        properties: Dict,
        parent_page_id: Optional[str] = None
    ) -> Dict:
        """
        Notion Database 생성
        
        Args:
            title: Database 제목
            properties: 필드 정의 (스키마)
            parent_page_id: 부모 페이지 ID (선택)
        
        Returns:
            생성된 Database 정보
        """
        parent_id = parent_page_id or self.parent_page_id
        
        try:
            response = self.client.databases.create(
                parent={
                    "type": "page_id",
                    "page_id": parent_id
                },
                title=[
                    {
                        "type": "text",
                        "text": {"content": title}
                    }
                ],
                properties=properties
            )
            
            print(f"✅ Database '{title}' 생성 완료!")
            print(f"   ID: {response['id']}")
            return response
            
        except Exception as e:
            print(f"❌ Database 생성 실패: {e}")
            raise
    
    def update_database_properties(
        self,
        database_id: str,
        properties: Dict
    ) -> Dict:
        """
        Database 속성(필드) 업데이트
        
        Args:
            database_id: Database ID
            properties: 추가/수정할 필드
        
        Returns:
            업데이트된 Database 정보
        """
        try:
            response = self.client.databases.update(
                database_id=database_id,
                properties=properties
            )
            
            print(f"✅ Database 속성 업데이트 완료!")
            return response
            
        except Exception as e:
            print(f"❌ 속성 업데이트 실패: {e}")
            raise
    
    def create_page(
        self,
        database_id: str,
        properties: Dict,
        children: Optional[List] = None
    ) -> Dict:
        """
        Database에 새 페이지 생성
        
        Args:
            database_id: Database ID
            properties: 페이지 속성 (필드 값)
            children: 페이지 내용 블록 (선택)
        
        Returns:
            생성된 페이지 정보
        """
        payload = {
            "parent": {"database_id": database_id},
            "properties": properties
        }
        
        if children:
            payload["children"] = children
        
        try:
            response = self.client.pages.create(**payload)
            print(f"✅ 페이지 생성 완료!")
            return response
            
        except Exception as e:
            print(f"❌ 페이지 생성 실패: {e}")
            raise
    
    def query_database(
        self,
        database_id: str,
        filter_obj: Optional[Dict] = None,
        sorts: Optional[List] = None
    ) -> List[Dict]:
        """
        Database 쿼리 (검색)
        
        Args:
            database_id: Database ID
            filter_obj: 필터 조건
            sorts: 정렬 조건
        
        Returns:
            검색된 페이지 리스트
        """
        payload = {"database_id": database_id}
        
        if filter_obj:
            payload["filter"] = filter_obj
        if sorts:
            payload["sorts"] = sorts
        
        try:
            response = self.client.databases.query(**payload)
            return response["results"]
            
        except Exception as e:
            print(f"❌ Database 쿼리 실패: {e}")
            raise
```

```python
import sys
import os

sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

from utils.notion_client import notion
```

```python
def get_stock_price(ticker: str) -> float:
    """
    네이버 금융에서 주가 가져오기
    
    Args:
        ticker: 종목코드 (예: 379800)
    
    Returns:
        현재가
    """
    url = f"https://finance.naver.com/item/main.nhn?code={ticker}"
    
    try:
        response = requests.get(url)
        soup = BeautifulSoup(response.text, 'html.parser')
        
        # 현재가 파싱
        price_element = soup.select_one('.no_today .blind')
        if price_element:
            price = price_element.text.replace(',', '')
            return float(price)
        
        return None
        
    except Exception as e:
        print(f"❌ 가격 가져오기 실패 ({ticker}): {e}")
        return None


def update_portfolio_prices():
    """
    Portfolio Holdings의 모든 자산 가격 업데이트
    """
    portfolio_db_id = os.getenv("PORTFOLIO_DB_ID")
    
    if not portfolio_db_id:
        print("❌ PORTFOLIO_DB_ID가 설정되지 않았습니다.")
        return
    
    print("=" * 60)
    print("📊 Portfolio 가격 업데이트 시작")
    print("=" * 60)
    
    # Portfolio Holdings 쿼리
    holdings = notion.query_database(portfolio_db_id)
    
    for holding in holdings:
        # 티커 가져오기
        ticker_prop = holding["properties"].get("티커/종목코드")
        if not ticker_prop or not ticker_prop.get("rich_text"):
            continue
        
        ticker = ticker_prop["rich_text"][0]["plain_text"]
        
        # 가격 가져오기
        print(f"\n{ticker} 가격 조회 중...")
        price = get_stock_price(ticker)
        
        if price:
            # Notion 업데이트
            notion.client.pages.update(
                page_id=holding["id"],
                properties={
                    "현재가": {"number": price}
                }
            )
            print(f"✅ 업데이트 완료: ₩{price:,.0f}")
        else:
            print(f"⚠️ 가격 조회 실패")
    
    print("\n" + "=" * 60)
    print("✅ 가격 업데이트 완료!")
    print("=" * 60)
```

```python
"""
Market Notes 백업
- 전체 DB 데이터를 JSON으로 추출
- GitHub Artifacts에 저장
"""

import os
import json
from notion_client import Client
from dotenv import load_dotenv
from datetime import datetime

load_dotenv()


def backup_database():
    """DB 백업"""
    client = Client(auth=os.getenv("NOTION_TOKEN"))
    db_id = os.getenv("MARKET_NOTES_DB_ID")
    
    print("💾 Backing up Market Notes...")
    
    # 전체 페이지 조회
    all_pages = []
    has_more = True
    start_cursor = None
    
    while has_more:
        response = client.databases.query(
            database_id=db_id,
            start_cursor=start_cursor
        )
        all_pages.extend(response["results"])
        has_more = response["has_more"]
        start_cursor = response.get("next_cursor")
    
    # 백업 파일 생성
    os.makedirs("backups", exist_ok=True)
    
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    filename = f"backups/market_notes_{timestamp}.json"
    
    backup_data = {
        "backup_time": datetime.now().isoformat(),
        "total_pages": len(all_pages),
        "database_id": db_id,
        "pages": all_pages
    }
    
    with open(filename, "w", encoding="utf-8") as f:
        json.dump(backup_data, f, ensure_ascii=False, indent=2)
    
    print(f"✅ Backup saved: {filename}")
    print(f"   Total pages: {len(all_pages)}")
    
    return filename


if __name__ == "__main__":
    backup_database()
```

```python
"""
거래 내역 자동 추가
GitHub Actions에서 수동 트리거
"""

import argparse
import os
from notion_client import Client
from datetime import datetime
from dotenv import load_dotenv

load_dotenv()


def add_transaction(asset: str, trans_type: str, amount: float, quantity: float, reason: str = ""):
    """거래 추가"""
    client = Client(auth=os.getenv("NOTION_TOKEN"))
    db_id = os.getenv("TRANSACTIONS_DB_ID")
    
    # 페이지 속성 구성
    properties = {
        "거래일": {
            "date": {"start": datetime.now().date().isoformat()}
        },
        "자산": {
            "title": [{"text": {"content": asset}}]
        },
        "거래 유형": {
            "select": {"name": trans_type}
        },
        "수량": {
            "number": quantity
        },
        "거래금액": {
            "number": amount
        },
        "거래 근거": {
            "rich_text": [{"text": {"content": reason or "수동 입력"}}]
        },
        "감정 상태": {
            "select": {"name": "😌 냉정"}
        }
    }
    
    # 페이지 생성
    client.pages.create(
        parent={"database_id": db_id},
        properties=properties
    )
    
    print(f"✅ Transaction added: {asset} - {trans_type} - ₩{amount:,.0f}")


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--asset", required=True)
    parser.add_argument("--type", required=True)
    parser.add_argument("--amount", type=float, required=True)
    parser.add_argument("--quantity", type=float, required=True)
    parser.add_argument("--reason", default="")
    
    args = parser.parse_args()
    
    add_transaction(
        asset=args.asset,
        trans_type=args.type,
        amount=args.amount,
        quantity=args.quantity,
        reason=args.reason
    )
```

```python
# utils/logger.py
import logging
from datetime import datetime

def setup_logger(name):
    """로거 설정"""
    logger = logging.getLogger(name)
    logger.setLevel(logging.INFO)
    
    # 파일 핸들러
    fh = logging.FileHandler(f'logs/{name}_{datetime.now():%Y%m%d}.log')
    fh.setLevel(logging.INFO)
    
    # 포맷터
    formatter = logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    )
    fh.setFormatter(formatter)
    
    logger.addHandler(fh)
    return logger


# 사용
```
