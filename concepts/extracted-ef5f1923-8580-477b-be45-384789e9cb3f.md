from anthropic import Anthropic
import os
from pathlib import Path

client = Anthropic(api_key=os.environ.get("ANTHROPIC_API_KEY"))

def automate_hr_task(folder_path, instruction):
    """
    Claude Code 방식의 자동화 함수
    """
    # 폴더 내 파일 목록 수집
    files = list(Path(folder_path).rglob("*.*"))
    context = "\n".join([f"- {f.name}" for f in files[:20]])
    
    message = client.messages.create(
        model="claude-sonnet-4-5-20250929",
        max_tokens=4096,
        messages=[{
            "role": "user",
            "content": f"""
작업 폴더: {folder_path}
파일 목록:
{context}

작업 지시: {instruction}

파일을 읽고, 분석하고, 필요한 작업을 수행하는 Python 코드를 작성해줘.
코드는 즉시 실행 가능해야 하며, 결과물을 저장해야 해.
"""
        }]
    )
    
    return message.content

# 사용 예시
result = automate_hr_task(

    instruction="모든 피드백을 통합하여 리더십 평가 종합 보고서를 만들어줘"
)
