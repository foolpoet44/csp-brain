    },

  };

    "innovation_culture": 0,
    "hierarchy_culture": 0,
    "market_culture": 0,
    "clan_culture": 0,
    "psychological_safety": 0
  };
  
  // 응답 집계

    Object.keys(scores).forEach(dimension => {
      cultureProfile[dimension] += scores[dimension];

  });

  return normalizeAndVisualize(cultureProfile);
}
