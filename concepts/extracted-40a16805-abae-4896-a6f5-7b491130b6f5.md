# Convergence/metrics/convergence_calculator.py

from dataclasses import dataclass
from typing import List, Dict

@dataclass
class Generation:
    gen_id: int
    ontology: Dict        # 생성된 결과물의 구조
    timestamp: str
    change_count: int

class ConvergenceCalculator:
    """
    Ouroboros 수렴 공식을 CSP 프로젝트에 맞게 구현
    Similarity = 0.5*name_overlap + 0.3*type_match + 0.2*exact_match
    """
    
    def __init__(self, config: dict):
        self.w_name = config.get('name_overlap_weight', 0.5)
        self.w_type = config.get('type_match_weight', 0.3)
        self.w_exact = config.get('exact_match_weight', 0.2)
        self.threshold = config.get('similarity_threshold', 0.95)
    
    def calculate_similarity(self, gen_prev: Generation, 
                              gen_curr: Generation) -> float:
        prev = gen_prev.ontology
        curr = gen_curr.ontology
        
        # 이름 겹침 (개념의 안정화)
        prev_names = set(prev.get('names', []))
        curr_names = set(curr.get('names', []))
        if not prev_names and not curr_names:
            name_overlap = 1.0
        else:
            name_overlap = len(prev_names & curr_names) / \
                          len(prev_names | curr_names)
        
        # 타입 일치 (구조의 안정화)
        prev_types = set(prev.get('types', []))
        curr_types = set(curr.get('types', []))
        if not prev_types and not curr_types:
            type_match = 1.0
        else:
            type_match = len(prev_types & curr_types) / \
                        len(prev_types | curr_types)
        
        # 정확 일치 (내용의 안정화)
        prev_items = set(str(i) for i in prev.get('items', []))
        curr_items = set(str(i) for i in curr.get('items', []))
        if not prev_items and not curr_items:
            exact_match = 1.0
        else:
            exact_match = len(prev_items & curr_items) / \
                         len(prev_items | curr_items)
        
        similarity = (self.w_name * name_overlap +
                     self.w_type * type_match +
                     self.w_exact * exact_match)
        
        return round(similarity, 4)
    
    def is_converged(self, gen_prev: Generation, 
                     gen_curr: Generation) -> bool:
        return self.calculate_similarity(gen_prev, gen_curr) >= \
               self.threshold

from enum import Enum
from typing import List, Optional

class PathologyType(Enum):
    STAGNATION = "stagnation"
    OSCILLATION = "oscillation"  
    REPETITIVE_FEEDBACK = "repetitive_feedback"
    WONDER_LOOP = "wonder_loop"
    HEALTHY = "healthy"

@dataclass
class PathologyResult:
    type: PathologyType
    severity: str          # "warning" | "critical"
    message: str
    recommended_action: str

class PathologyDetector:
    
    def detect(self, generations: List[Generation], 
               feedbacks: List[str]) -> PathologyResult:
        
        # 1. Stagnation 감지
        if len(generations) >= 3:
            last_three = generations[-3:]
            if all(g.change_count == 0 for g in last_three):
                return PathologyResult(
                    type=PathologyType.STAGNATION,
                    severity="critical",
                    message="3세대 연속 변화 없음. AI가 막혀있습니다.",
                    recommended_action="프롬프트에 새로운 제약조건 또는 "
                                      "예시를 추가하세요."
                )
        
        # 2. Oscillation 감지
        if len(generations) >= 4:
            gen_n = generations[-1].ontology
            gen_n2 = generations[-3].ontology
            if gen_n == gen_n2:
                return PathologyResult(
                    type=PathologyType.OSCILLATION,
                    severity="critical", 
                    message="Gen N과 Gen N-2가 동일. 두 안 사이 진동 중.",
                    recommended_action="강제로 한쪽 방향을 선택하는 "
                                      "제약을 추가하세요."
                )
        
        # 3. Repetitive Feedback 감지
        if len(feedbacks) >= 3:
            overlap = self._calculate_text_overlap(feedbacks[-3:])
            if overlap >= 0.70:
                return PathologyResult(
                    type=PathologyType.REPETITIVE_FEEDBACK,
                    severity="warning",
                    message=f"피드백 겹침 {overlap:.0%}. 같은 질문 반복.",
                    recommended_action="초기 온톨로지(맥락) 주입을 "
                                      "보강하세요."
                )
        
        # 4. Wonder Loop 감지
        if len(feedbacks) >= 5:
            keywords = self._extract_keywords(feedbacks)
            top_keyword_ratio = self._top_keyword_ratio(keywords)
            if top_keyword_ratio >= 0.30:
                return PathologyResult(
                    type=PathologyType.WONDER_LOOP,
                    severity="warning",
                    message="탐색 공간 고착. 같은 키워드 중심 반복.",
                    recommended_action="탐색 다양성을 높이는 "
                                      "temperature 조정 또는 "
                                      "도메인 재분할을 시도하세요."
                )
        
        return PathologyResult(
            type=PathologyType.HEALTHY,
            severity="ok",
            message="정상 진화 중",
            recommended_action="continue"
        )
    
    def _calculate_text_overlap(self, texts: List[str]) -> float:
        # 텍스트 간 평균 자카드 유사도
        from itertools import combinations
        if len(texts) < 2:
            return 0.0
        scores = []
        for a, b in combinations(texts, 2):
            set_a = set(a.split())
            set_b = set(b.split())
            if not set_a and not set_b:
                scores.append(1.0)
            else:
                scores.append(len(set_a & set_b) / len(set_a | set_b))
        return sum(scores) / len(scores)

## Convergence Control

모든 반복 작업은 Convergence Layer를 통해 제어된다.

### 필수 절차
1. 작업 시작 전: `Convergence/dod/` 에서 해당 프로젝트 DoD 로드
2. 매 이터레이션 후: `ConvergenceCalculator`로 유사도 계산

- Pathology severity == "critical" → 즉시 중단, 인간 에스컬레이션
- Generation count >= max_generations → 강제 중단, 로그 기록

### 재개 조건
- 인간이 새로운 컨텍스트를 주입한 경우에만 재시작 가능
