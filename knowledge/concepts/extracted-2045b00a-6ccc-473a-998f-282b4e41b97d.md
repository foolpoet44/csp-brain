from pptx import Presentation
from pptx.util import Inches, Pt
from gtts import gTTS
from moviepy.editor import *
from PIL import Image
import os
from datetime import datetime

class EducationVideoAutomator:
    """교육 영상 자동 생성 클래스"""
    
    def __init__(self, output_dir="./output"):
        self.output_dir = output_dir
        os.makedirs(output_dir, exist_ok=True)
        
    def create_slide_deck(self, content_data):
        """
        Notion 데이터로 PPT 자동 생성
        
        content_data 구조:
        {
            'title': '리더십의 본질',
            'slides': [
                {
                    'title': '슬라이드 제목',
                    'content': '슬라이드 내용',
                    'narration': '나레이션 스크립트'
                },
                ...
            ]
        }
        """
        prs = Presentation()
        prs.slide_width = Inches(16)
        prs.slide_height = Inches(9)
        
        # 제목 슬라이드
        title_slide = prs.slides.add_slide(prs.slide_layouts[0])
        title_slide.shapes.title.text = content_data['title']
        subtitle = title_slide.placeholders[1]
        subtitle.text = f"생성일: {datetime.now().strftime('%Y-%m-%d')}\n제작: HR자동화시스템"
        
        # 내용 슬라이드들
        for slide_data in content_data['slides']:
            slide = prs.slides.add_slide(prs.slide_layouts[1])
            
            # 제목
            slide.shapes.title.text = slide_data['title']
            
            # 본문
            content_box = slide.placeholders[1]
            content_box.text = slide_data['content']
            
            # 폰트 설정 (한국어 가독성)
            for paragraph in content_box.text_frame.paragraphs:
                for run in paragraph.runs:
                    run.font.name = '맑은 고딕'
                    run.font.size = Pt(18)
        
        ppt_path = f"{self.output_dir}/{content_data['title']}.pptx"
        prs.save(ppt_path)
        print(f"✅ PPT 생성 완료: {ppt_path}")
        return ppt_path
    
    def generate_narrations(self, content_data):
        """
        각 슬라이드의 나레이션 음성 파일 생성
        """
        audio_files = []
        
        for idx, slide_data in enumerate(content_data['slides']):
            narration_text = slide_data.get('narration', slide_data['content'])
            
            # gTTS로 음성 생성 (한국어)
            tts = gTTS(text=narration_text, lang='ko', slow=False)
            
            audio_path = f"{self.output_dir}/narration_{idx}.mp3"
            tts.save(audio_path)
            audio_files.append(audio_path)
            
            print(f"🎙️ 나레이션 생성: 슬라이드 {idx+1}")
        
        return audio_files
    
    def ppt_to_images(self, ppt_path):
        """
        PPT를 이미지로 변환
        (실제로는 LibreOffice 커맨드라인 사용)
        """
        # LibreOffice headless 모드로 PPT → PDF → 이미지
        pdf_path = ppt_path.replace('.pptx', '.pdf')
        
        # PPT → PDF 변환
        os.system(f'libreoffice --headless --convert-to pdf --outdir {self.output_dir} {ppt_path}')
        
        # PDF → 이미지 변환 (ImageMagick 사용)
        image_dir = f"{self.output_dir}/slides"
        os.makedirs(image_dir, exist_ok=True)
        os.system(f'convert -density 300 {pdf_path} {image_dir}/slide_%03d.png')
        
        # 이미지 파일 목록 반환
        images = sorted([f"{image_dir}/{f}" for f in os.listdir(image_dir) if f.endswith('.png')])
        print(f"🖼️ 슬라이드 이미지 생성: {len(images)}개")
        return images
    
    def create_video(self, image_files, audio_files, content_data):
        """
        이미지 + 음성 → 영상 합성
        """
        clips = []
        
        for idx, (image_path, audio_path) in enumerate(zip(image_files, audio_files)):
            # 음성 길이 계산
            audio_clip = AudioFileClip(audio_path)
            duration = audio_clip.duration
            
            # 이미지 클립 생성
            image_clip = ImageClip(image_path).set_duration(duration)
            
            # 음성 추가
            video_clip = image_clip.set_audio(audio_clip)
            
            clips.append(video_clip)
            print(f"🎬 슬라이드 {idx+1} 처리 완료 ({duration:.1f}초)")
        
        # 모든 클립 연결
        final_video = concatenate_videoclips(clips, method="compose")
        
        # 인트로/아웃트로 추가 (선택사항)
        # final_video = self.add_intro_outro(final_video, content_data)
        
        # 영상 저장
        video_path = f"{self.output_dir}/{content_data['title']}_교육영상.mp4"
        final_video.write_videofile(
            video_path,
            fps=24,
            codec='libx264',
            audio_codec='aac',
            temp_audiofile='temp-audio.m4a',
            remove_temp=True
        )
        
        print(f"✅ 최종 영상 생성 완료: {video_path}")
        return video_path
    
    def run(self, content_data):
        """
        전체 파이프라인 실행
        """
        print("🚀 교육 영상 자동 생성 시작...")
        
        # 1. PPT 생성
        ppt_path = self.create_slide_deck(content_data)
        
        # 2. 나레이션 생성
        audio_files = self.generate_narrations(content_data)
        
        # 3. PPT → 이미지
        image_files = self.ppt_to_images(ppt_path)
        
        # 4. 영상 합성
        video_path = self.create_video(image_files, audio_files, content_data)
        
        print(f"""
        ✨ 완료! ✨
        
        생성된 파일:
        - PPT: {ppt_path}
        - 영상: {video_path}
        
        총 소요 시간: 약 3-5분
        CSP님이 한 일: 0번 클릭
        """)
        
        return video_path

    def _extract_ppt_content(self, ppt_path: str) -> Dict:
        """PPT 내용 추출 (간단 버전)"""
        from pptx import Presentation
        
        prs = Presentation(ppt_path)
        content = {
            'title': prs.slides[0].shapes.title.text if prs.slides else "Untitled",
            'slides': []
        }
        
        for slide in prs.slides:
            slide_data = {
                'title': slide.shapes.title.text if slide.shapes.title else "",
                'content': self._extract_text_from_shapes(slide.shapes)
            }
            content['slides'].append(slide_data)
        
        return content
    
    def _extract_text_from_shapes(self, shapes):
        """슬라이드에서 텍스트 추출"""
        text_parts = []
        for shape in shapes:
            if hasattr(shape, "text"):
                text_parts.append(shape.text)
        return "\n".join(text_parts)
