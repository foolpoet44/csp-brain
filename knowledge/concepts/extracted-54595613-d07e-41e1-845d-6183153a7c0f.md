import re
import yaml
from pathlib import Path
from enum import Enum
from typing import Dict, Any, List
from loguru import logger


class SecurityLevel(str, Enum):
    """보안 레벨"""
    PUBLIC = "PUBLIC"
    INTERNAL = "INTERNAL"
    CONFIDENTIAL = "CONFIDENTIAL"


class SecurityFilter:
    """보안 필터링 클래스"""
    
    def __init__(self, rules_path: str = "config/security_rules.yaml"):
        """
        Args:
            rules_path: 보안 규칙 YAML 파일 경로
        """
        self.rules_path = Path(rules_path)
        self.rules = self._load_rules()
        logger.info(f"Security rules loaded from {rules_path}")
    
    def _load_rules(self) -> Dict[str, Any]:
        """보안 규칙 YAML 로드"""
        if not self.rules_path.exists():
            logger.warning(f"Rules file not found: {self.rules_path}")
            return self._get_default_rules()
        
        with open(self.rules_path, 'r', encoding='utf-8') as f:
            return yaml.safe_load(f)
    
    def _get_default_rules(self) -> Dict[str, Any]:
        """기본 보안 규칙"""
        return {
            'security_levels': {
                'PUBLIC': {'mask_all_numbers': True}
            },
            'patterns': {},
            'sensitive_keywords': []
        }
    
    def apply_filters(
        self, 
        text: str, 
        level: SecurityLevel = SecurityLevel.INTERNAL
    ) -> str:
        """
        텍스트에 보안 필터 적용
        
        Args:
            text: 원본 텍스트
            level: 보안 레벨
            
        Returns:
            필터링된 텍스트
        """
        if not text:
            return text
        
        level_config = self.rules['security_levels'].get(level.value, {})
        filtered_text = text
        
        # 숫자 마스킹
        if level_config.get('mask_all_numbers', False):
            filtered_text = self._mask_all_patterns(filtered_text)
        
        # 조직명 마스킹
        if level_config.get('mask_organizations', False):
            filtered_text = self._mask_organizations(filtered_text)
        
        # 날짜 마스킹
        if level_config.get('mask_dates', False):
            filtered_text = self._mask_dates(filtered_text)
        
        # 이름 마스킹
        if level_config.get('mask_names', False):
            filtered_text = self._mask_names(filtered_text)
        
        return filtered_text
    
    def _mask_all_patterns(self, text: str) -> str:
        """모든 패턴 마스킹 적용"""
        result = text
        patterns = self.rules.get('patterns', {})
        
        for category, rules in patterns.items():
            for rule in rules:
                regex = rule.get('regex')
                replacement = rule.get('replacement')
                if regex and replacement:
                    result = re.sub(regex, replacement, result)
        
        return result
    
    def _mask_organizations(self, text: str) -> str:
        """조직명 마스킹"""

            result = re.sub(pattern, replacement, result)

            replacement = rule.get('replacement')

        result = re.sub(r'\b([가-힣]{2,4})\s*(님|씨|대표|이사|부장|팀장)', r'***\2', text)

            left = self.MARGIN_LEFT + (i * (table_width + Inches(0.3)))

            cols = len(table_data.Columns)

                rows, cols, left, top, table_width, height
            ).table

            for col_idx, col_name in enumerate(table_data.Columns):
                cell = table.cell(0, col_idx)
                cell.text = col_name
                cell.text_frame.paragraphs[0].font.size = Pt(10)
                cell.text_frame.paragraphs[0].font.bold = True
                cell.fill.solid()
                cell.fill.fore_color.rgb = RGBColor(230, 230, 230)

                for col_idx, cell_text in enumerate(row_data):
                    cell = table.cell(row_idx + 1, col_idx)
                    cell.text = str(cell_text)
                    cell.text_frame.paragraphs[0].font.size = Pt(9)

        """테이블 상하 배치"""

            json_path: JSON 파일 경로
            
        Returns:

        """
        path = Path(json_path)

        try:
            with open(path, 'r', encoding='utf-8') as f:

            # Pydantic 스키마 검증

            logger.success(f"Schema validation passed for {spec.DeckName}")
            
            return spec
            
        except json.JSONDecodeError as e:
            logger.error(f"Invalid JSON format: {e}")
            return None
        
        except ValidationError as e:
            logger.error(f"Schema validation failed:\n{e}")
            return None
        
        except Exception as e:
            logger.error(f"Unexpected error: {e}")
            return None

        """
        스키마 검증만 수행
        
        Args:
            data: 딕셔너리 데이터
            
        Returns:
            검증 성공 여부
        """

            SlideFormatSpec(**data)
            return True
        except ValidationError:
            return False
    
    def extract_metadata(self, spec: SlideFormatSpec) -> dict:
        """
        메타데이터 추출
        
        Args:
            spec: SlideFormatSpec 객체
            
        Returns:
            메타데이터 딕셔너리
        """
        slide_types = {}
        for slide in spec.Slides:
            slide_type = slide.SlideType.value
            slide_types[slide_type] = slide_types.get(slide_type, 0) + 1
        
        metadata = {

            'total_slides': len(spec.Slides),
            'slide_types': slide_types,

from loguru import logger

from src.core.schema import SlideFormatSpec, Slide
from src.core.security import SecurityFilter, SecurityLevel


class SecurityAgent:
    """보안 필터링 에이전트"""
    
    def __init__(self, rules_path: str = "config/security_rules.yaml"):
        """
        Args:
            rules_path: 보안 규칙 파일 경로
        """
        self.filter = SecurityFilter(rules_path)
        self.report = {
            'filtered_count': 0,
            'slides_affected': []
        }
        logger.info("SecurityAgent initialized")
    
    def filter_deck(
        self,
        spec: SlideFormatSpec,
        level: SecurityLevel = SecurityLevel.INTERNAL
    ) -> SlideFormatSpec:
        """
        전체 덱 필터링
        
        Args:
            spec: 원본 SlideFormatSpec
            level: 보안 레벨
            
        Returns:
            필터링된 SlideFormatSpec
        """
        logger.info(f"Filtering deck with {level.value} level")
        
        filtered_slides = []
        for slide in spec.Slides:
            filtered_slide = self.filter_slide(slide, level)
            filtered_slides.append(filtered_slide)
        
        filtered_spec = SlideFormatSpec(
            DeckName=spec.DeckName,
            Slides=filtered_slides
        )
        
        logger.success(f"Deck filtering completed: {self.report['filtered_count']} items filtered")
        return filtered_spec
    
    def filter_slide(self, slide: Slide, level: SecurityLevel) -> Slide:
        """
        개별 슬라이드 필터링
        
        Args:
            slide: 원본 Slide
            level: 보안 레벨
            
        Returns:
            필터링된 Slide
        """
        slide_dict = slide.model_dump()
        filtered_dict = self._filter_recursive(slide_dict, level)
        
        return Slide(**filtered_dict)
    
    def _filter_recursive(self, obj: Any, level: SecurityLevel) -> Any:
        """재귀적 필터링"""
        if isinstance(obj, dict):
            return {k: self._filter_recursive(v, level) for k, v in obj.items()}
        elif isinstance(obj, list):
            return [self._filter_recursive(item, level) for item in obj]
        elif isinstance(obj, str):
            filtered = self.filter.apply_filters(obj, level)
            if filtered != obj:
                self.report['filtered_count'] += 1
            return filtered
        else:
            return obj
    
    def generate_report(self) -> Dict[str, Any]:
        """
        필터링 리포트 생성
        
        Returns:
            리포트 딕셔너리
        """
        return {
            'total_filtered': self.report['filtered_count'],
            'slides_affected': len(self.report['slides_affected']),
            'summary': f"{self.report['filtered_count']} items were filtered"
        }
