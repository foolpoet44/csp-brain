# skill_loader.py
class SkillLoader:
    def __init__(self, skills_dir: Path):
        self.skills_dir = skills_dir
        self.loaded_skills = {}
        
    def scan_metadata(self, query: str) -> List[str]:
        """
        Progressive Disclosure Step 1
        모든 스킬의 메타데이터만 읽어서 관련성 판단
        """
        relevant_skills = []
        for skill_path in self.skills_dir.glob("*/SKILL.md"):
            metadata = self.extract_frontmatter(skill_path)
            if self.is_relevant(metadata, query):
                relevant_skills.append(metadata['name'])
        return relevant_skills
    
    def load_full_skill(self, skill_name: str) -> Skill:
        """
        Progressive Disclosure Step 2
        선택된 스킬만 전체 로드
        """
        skill_path = self.skills_dir / skill_name / "SKILL.md"
        return Skill.from_markdown(skill_path)
    
    def is_relevant(self, metadata: dict, query: str) -> bool:
        """
        임베딩 기반 유사도 또는 키워드 매칭
        """
        triggers = metadata.get('triggers', [])
        description = metadata.get('description', '')
        
        # 간단한 키워드 매칭 (실제로는 임베딩 사용)
        for trigger in triggers:
            if trigger.lower() in query.lower():
                return True
        return False
