async function processDailyTasks() {
    const today = moment().format('YYYY-MM-DD');
    const yesterday = moment().subtract(1, 'day').format('YYYY-MM-DD');
    
    // 어제 노트에서 미완료 작업 추출
    const yesterdayFile = app.vault.getAbstractFileByPath(`Daily Notes/${yesterday}.md`);
    if (!yesterdayFile) return;
    
    const yesterdayContent = await app.vault.read(yesterdayFile);
    const uncompletedTasks = extractUncompletedTasks(yesterdayContent);
    
    // 오늘 노트에 이월 작업 추가
    const todayFile = app.vault.getAbstractFileByPath(`Daily Notes/${today}.md`);
    if (todayFile) {
        let todayContent = await app.vault.read(todayFile);
        
        if (uncompletedTasks.length > 0) {
            const carryoverSection = generateCarryoverSection(uncompletedTasks);
            todayContent = todayContent.replace(
                '## 🎯 Today\'s Priorities',
                `## 🎯 Today's Priorities\n\n### 🔄 이월된 작업\n${carryoverSection}\n`
            );
            
            await app.vault.modify(todayFile, todayContent);
            
            // 알림 표시
            new Notice(`${uncompletedTasks.length}개의 작업이 이월되었습니다.`);
        }
    }
    
    // 프로젝트 상태 업데이트
    await updateProjectStatus();
    
    // 주간/월간 요약이 필요한지 확인
    await checkPeriodicalReports();
}

function extractUncompletedTasks(content) {
    const tasks = [];
    const lines = content.split('\n');
    
    for (let line of lines) {
        // 미완료 작업 패턴: - [ ] 텍스트
        const match = line.match(/^- \[ \] (.+)$/);
        if (match && !line.includes('#routine')) {
            tasks.push({
                text: match[1],
                priority: line.includes('#urgent') ? 'urgent' : 
                         line.includes('#important') ? 'important' : 'normal',
                tags: extractTags(line)
            });
        }
    }
    
    return tasks;
}

function generateCarryoverSection(tasks) {
    return tasks.map(task => 
        `- [ ] ${task.text} #carryover ${task.tags.join(' ')}`
    ).join('\n');
}

async function updateProjectStatus() {
    // 프로젝트 폴더의 모든 프로젝트 파일 스캔
    const projectFiles = app.vault.getMarkdownFiles()
        .filter(file => file.path.startsWith('Projects/'));
        
    for (let file of projectFiles) {
        const content = await app.vault.read(file);
        const frontmatter = app.metadataCache.getFileCache(file)?.frontmatter;
        
        if (frontmatter) {
            // 작업 완료율 계산
            const tasks = extractTasksFromContent(content);
            const completionRate = calculateCompletionRate(tasks);
            
            // 마지막 업데이트 날짜 확인
            const lastUpdate = frontmatter['last-update'];
            const today = moment().format('YYYY-MM-DD');
            
            if (lastUpdate !== today && completionRate !== frontmatter.progress) {
                // 프론트매터 업데이트
                await updateProjectFrontmatter(file, {
                    progress: completionRate,
                    'last-update': today,
                    'tasks-total': tasks.length,
                    'tasks-completed': tasks.filter(t => t.completed).length
                });
            }
        }
    }
}

class ProjectAnalyzer {
    constructor() {
        this.app = app;
    }
    
    async analyzeAllProjects() {
        const projects = await this.getAllProjects();
        const analysis = {
            summary: await this.generateSummary(projects),
            risks: await this.identifyRisks(projects),
            recommendations: await this.generateRecommendations(projects),
            trends: await this.analyzeTrends(projects)
        };
        
        await this.generateAnalysisReport(analysis);
        return analysis;
    }
    
    async getAllProjects() {
        const projectFiles = this.app.vault.getMarkdownFiles()
            .filter(file => file.path.startsWith('Projects/'));
            
        const projects = [];
        for (let file of projectFiles) {
            const content = await this.app.vault.read(file);
            const metadata = this.app.metadataCache.getFileCache(file)?.frontmatter || {};
            const tasks = this.extractTasks(content);
            
            projects.push({
                name: metadata.title || file.basename,
                file: file,
                status: metadata.status,
                priority: metadata.priority,
                progress: metadata.progress || 0,
                dueDate: metadata.due,
                tasks: tasks,
                lastUpdate: metadata['last-update'],
                tags: metadata.tags || [],
                ...metadata
            });
        }
        
        return projects;
    }
    
    async identifyRisks(projects) {
        const risks = [];
        
        for (let project of projects) {
            // 지연 위험
            if (project.dueDate && moment(project.dueDate).isBefore(moment(), 'day')) {
                if (project.progress < 100) {
                    risks.push({
                        type: 'overdue',
                        severity: 'high',
                        project: project.name,
                        message: `프로젝트가 ${moment().diff(project.dueDate, 'days')}일 지연됨`
                    });
                }
            }
            
            // 진행률 위험
            if (project.dueDate) {
                const totalDays = moment(project.dueDate).diff(moment(project.startDate || project.created), 'days');
                const passedDays = moment().diff(moment(project.startDate || project.created), 'days');
                const expectedProgress = Math.min((passedDays / totalDays) * 100, 100);
                
                if (project.progress < expectedProgress - 20) {
                    risks.push({
                        type: 'behind_schedule',
                        severity: 'medium',
                        project: project.name,
                        message: `예상 진행률(${Math.round(expectedProgress)}%)보다 ${Math.round(expectedProgress - project.progress)}% 뒤처짐`
                    });
                }
            }
            
            // 블로커 위험
            const blockers = project.tasks.filter(task => 
                task.text.includes('#blocked') || task.text.includes('#waiting')
            );
            if (blockers.length >= 3) {
                risks.push({
                    type: 'too_many_blockers',
                    severity: 'medium',
                    project: project.name,
                    message: `${blockers.length}개의 블로킹 이슈 존재`
                });
            }
        }
        
        return risks;
    }
    
    async generateAnalysisReport(analysis) {
        const reportContent = `---
type: analysis-report
generated: ${moment().format('YYYY-MM-DD HH:mm:ss')}
---

# 📊 프로젝트 분석 리포트
*Generated: ${moment().format('YYYY년 MM월 DD일 HH:mm:ss')}*

## 🎯 전체 현황
- **활성 프로젝트**: ${analysis.summary.active}개
- **평균 진행률**: ${analysis.summary.avgProgress}%
- **위험 프로젝트**: ${analysis.risks.filter(r => r.severity === 'high').length}개

## ⚠️ 위험 요소

${analysis.risks.map(risk => `
### ${risk.severity === 'high' ? '🚨' : '⚠️'} ${risk.project}
**위험 유형**: ${risk.type}
**심각도**: ${risk.severity}
**내용**: ${risk.message}

`).join('')}

## 📈 권장 사항

${analysis.recommendations.map(rec => `- ${rec}`).join('\n')}

## 📊 트렌드 분석

### 이번 주 완성된 작업
\`\`\`chart
type: bar
data: ${JSON.stringify(analysis.trends.weeklyCompletion)}
\`\`\`

### 프로젝트별 진행률
\`\`\`chart
type: radar
data: ${JSON.stringify(analysis.trends.projectProgress)}
\`\`\`

---
*다음 분석 예정일: ${moment().add(1, 'week').format('YYYY-MM-DD')}*
`;

        const reportFile = `Reports/Project Analysis ${moment().format('YYYY-MM-DD')}.md`;
        await this.app.vault.create(reportFile, reportContent);
        
        new Notice('프로젝트 분석 리포트가 생성되었습니다!');
    }
}

class TemplateManager {
    async updateTemplate(templateName, changes) {
        const templateFile = app.vault.getAbstractFileByPath(`Templates/${templateName}.md`);
        if (!templateFile) {
            new Notice('템플릿을 찾을 수 없습니다.');
            return;
        }
        
        let content = await app.vault.read(templateFile);
        
        // 변경사항 적용
        for (let change of changes) {
            switch (change.type) {
                case 'add_section':
                    content = this.addSection(content, change.section, change.position);
                    break;
                case 'modify_query':
                    content = this.modifyDataviewQuery(content, change.oldQuery, change.newQuery);
                    break;
                case 'add_button':
                    content = this.addButton(content, change.button);
                    break;
                case 'update_frontmatter':
                    content = this.updateFrontmatter(content, change.fields);
                    break;
            }
        }
        
        // 백업 생성
        const backupPath = `Templates/Backups/${templateName}-${moment().format('YYYY-MM-DD-HHmm')}.md`;
        await app.vault.create(backupPath, await app.vault.read(templateFile));
        
        // 새 내용으로 업데이트
        await app.vault.modify(templateFile, content);
        
        new Notice(`템플릿 '${templateName}'이 업데이트되었습니다.`);
        
        // 변경 로그 기록
        await this.logTemplateChange(templateName, changes);
    }
    
    addSection(content, section, position) {
        const lines = content.split('\n');
        let insertIndex;
        
        if (position.type === 'after_header') {
            insertIndex = lines.findIndex(line => 
                line.startsWith('#') && line.includes(position.header)
            ) + 1;
        } else if (position.type === 'before_section') {
            insertIndex = lines.findIndex(line => 
                line.includes(position.marker)
            );
        }
        
        if (insertIndex >= 0) {
            lines.splice(insertIndex, 0, '', section, '');
        }
        
        return lines.join('\n');
    }
    
    modifyDataviewQuery(content, oldQuery, newQuery) {
        return content.replace(oldQuery, newQuery);
    }
    
    async logTemplateChange(templateName, changes) {
        const logPath = 'Automation/Logs/template-changes.md';
        const logEntry = `
## ${moment().format('YYYY-MM-DD HH:mm:ss')} - ${templateName}

${changes.map(change => `- **${change.type}**: ${change.description || 'No description'}`).join('\n')}

---
`;
        
        let logContent = '';
        const logFile = app.vault.getAbstractFileByPath(logPath);
        if (logFile) {
            logContent = await app.vault.read(logFile);
        } else {
            logContent = '# 템플릿 변경 로그\n\n';
        }
        
        const newLogContent = logContent.replace('# 템플릿 변경 로그\n\n', `# 템플릿 변경 로그\n\n${logEntry}`);
        
        if (logFile) {
            await app.vault.modify(logFile, newLogContent);
        } else {
            await app.vault.create(logPath, newLogContent);
        }
    }
}
