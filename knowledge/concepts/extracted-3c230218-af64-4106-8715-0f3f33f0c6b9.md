class PulseResponse(BaseModel):
    token: str
    q1: int
    q2: int
    q3: int
    q4: str = ""
    q5: str

# 응답 페이지 제공

@app.post("/api/submit")
async def submit_response(response: PulseResponse):
    # 토큰으로 직원 정보 조회
    users_sheet = gc.open('Pulse_Survey').worksheet('users')
    user_data = users_sheet.get_all_records()
    
    user = next((u for u in user_data if u['token'] == response.token), None)
    if not user:
        raise HTTPException(status_code=400, detail="Invalid token")
    
    # 응답 저장
    week_id = datetime.now().strftime('%Y-W%U')
    row = [
        datetime.now().isoformat(),
        user['employee_id'],
        user['department'],
        week_id,
        response.q1,
        response.q2,
        response.q3,
        response.q4,
        response.q5
    ]
    sheet.append_row(row)
    
    return {"status": "success"}

# 집계 API (대시보드용)
@app.get("/api/results/{week_id}")
async def get_results(week_id: str, department: str = None):
    records = sheet.get_all_records()
    
    # 필터링
    filtered = [r for r in records if r['week_id'] == week_id]
    if department:
        filtered = [r for r in filtered if r['department'] == department]
    
    if not filtered:
        return {"error": "No data"}
    
    # 집계
    return {
        "total_responses": len(filtered),
        "avg_q1": sum(r['q1'] for r in filtered) / len(filtered),
        "avg_q2": sum(r['q2'] for r in filtered) / len(filtered),
        "avg_q3": sum(r['q3'] for r in filtered) / len(filtered),
        "comments": [r['q4'] for r in filtered if r['q4']],
        "moods": [r['q5'] for r in filtered]
    }
