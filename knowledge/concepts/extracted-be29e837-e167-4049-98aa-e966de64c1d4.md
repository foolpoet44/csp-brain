```python
# 필요 라이브러리
# pip install google-cloud-texttospeech moviepy python-pptx

from google.cloud import texttospeech
from moviepy.editor import *
import os

# 1. 구글 슬라이드 발표자 노트 추출 (수동 또는 API)
scripts = [
    "슬라이드 1 스크립트...",
    "슬라이드 2 스크립트...",
    # ...
]

# 2. Google TTS로 음성 생성
client = texttospeech.TextToSpeechClient()

audio_files = []
for i, script in enumerate(scripts):
    synthesis_input = texttospeech.SynthesisInput(text=script)
    voice = texttospeech.VoiceSelectionParams(
        language_code="ko-KR",
        name="ko-KR-Wavenet-A"
    )
    audio_config = texttospeech.AudioConfig(
        audio_encoding=texttospeech.AudioEncoding.MP3
    )
    
    response = client.synthesize_speech(
        input=synthesis_input, 
        voice=voice, 
        audio_config=audio_config
    )
    
    filename = f"slide_{i}.mp3"
    with open(filename, "wb") as out:
        out.write(response.audio_content)
    audio_files.append(filename)

# 3. 슬라이드 이미지 + 음성 결합
from moviepy.editor import ImageClip, AudioFileClip, concatenate_videoclips

clips = []
for i, audio_file in enumerate(audio_files):
    audio = AudioFileClip(audio_file)
    image = ImageClip(f"slide_{i}.png").set_duration(audio.duration)
    video = image.set_audio(audio)
    clips.append(video)

# 4. 최종 동영상 생성
final_video = concatenate_videoclips(clips, method="compose")
final_video.write_videofile("presentation.mp4", fps=24)
```
