```bash
#!/bin/bash

TASK="React로 할일 관리 앱 만들기"

while true; do
    # 1단계: 현재 상태 확인
    CURRENT_STATE=$(git log --oneline -10)
    PROJECT_FILES=$(ls -R)
    
    # 2단계: AI에게 질문
    RESPONSE=$(claude_api \
        "Task: $TASK
         Current git log: $CURRENT_STATE
         Current files: $PROJECT_FILES
         
         다음 할 작업을 수행하고, 완료 여부를 알려줘.
         완료되면 'TASK_COMPLETE'라고 응답해줘.")
    
    # 3단계: 완료 확인
    if echo "$RESPONSE" | grep -q "TASK_COMPLETE"; then
        echo "작업 완료!"
        break
    fi
    
    # 4단계: 진행 상황 저장
    git add .
    git commit -m "AI 작업 진행: $(date)"
    
    # 5단계: 다음 반복으로
    sleep 5
done
```
