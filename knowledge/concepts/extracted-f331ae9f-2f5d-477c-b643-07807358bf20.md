# ══════════════════════════════════════════════════════════
#  프로젝트: 부서별 스킬 갭 분석 시스템
#  목표: 현재 보유 스킬 vs 시장 요구 스킬 비교
# ══════════════════════════════════════════════════════════

import pandas as pd
from collections import Counter, defaultdict
from skills_ml.ontologies.onet import Onet
from skills_ml.algorithms.skill_extractors import ExactMatchSkillExtractor

# ══════════════════════════════════════════
#  PART 1: 데이터 준비
# ══════════════════════════════════════════

class SkillGapAnalyzer:
    """스킬 갭 분석 클래스"""
    
    def __init__(self):
        print("시스템 초기화 중...")
        self.onet = Onet()
        self.extractor = ExactMatchSkillExtractor(
            self.onet.competency_framework
        )
        print("✓ 초기화 완료\n")
    
    def load_employee_skills(self, csv_file):
        """
        직원 스킬 데이터 로드
        
        CSV 형식:
        employee_id, name, department, skills
        E001, 홍길동, 개발팀, "Python,Java,SQL"
        """
        print(f"직원 데이터 로딩: {csv_file}")
        df = pd.read_csv(csv_file)
        
        # 스킬 문자열을 리스트로 변환
        df['skills_list'] = df['skills'].str.split(',')
        df['skills_list'] = df['skills_list'].apply(
            lambda x: [s.strip() for s in x]
        )
        
        print(f"✓ {len(df)}명의 직원 데이터 로드 완료\n")
        return df
    
    def load_market_requirements(self, job_postings):
        """
        시장 요구사항 분석 (채용공고 기반)
        
        job_postings: 채용공고 딕셔너리 리스트
        """
        print("시장 요구사항 분석 중...")
        market_skills = Counter()
        
        for job in job_postings:
            for candidate_skill in self.extractor.candidate_skills(job):
                market_skills[candidate_skill.skill_name] += 1
        
        print(f"✓ {len(market_skills)}개 스킬 발견\n")
        return market_skills
    
    def analyze_department_gap(self, employee_df, market_skills, department):
        """부서별 스킬 갭 분석"""
        
        print("="*70)
        print(f"부서: {department}")
        print("="*70)
        
        # 해당 부서 직원 필터링
        dept_employees = employee_df[
            employee_df['department'] == department
        ]
        
        # 부서 보유 스킬 집계
        dept_skills = Counter()
        for skills_list in dept_employees['skills_list']:
            dept_skills.update(skills_list)
        
        # 시장 요구 상위 20개 스킬
        top_market_skills = dict(market_skills.most_common(20))
        
        # 갭 분석
        print(f"\n부서 인원: {len(dept_employees)}명")
        print(f"부서 보유 스킬 종류: {len(dept_skills)}개\n")
        
        print("스킬 갭 분석 결과:")
        print("-"*70)
        print(f"{'스킬':<25} {'시장수요':>10} {'보유인원':>10} {'갭':>10}")
        print("-"*70)
        
        gap_analysis = []
        
        for skill, market_demand in top_market_skills.items():
            current_count = dept_skills.get(skill, 0)
            gap = market_demand - current_count
            
            gap_analysis.append({
                'skill': skill,
                'market_demand': market_demand,
                'current_count': current_count,
                'gap': gap,
                'coverage': (current_count / market_demand * 100) 
                           if market_demand > 0 else 0
            })
        
        # 갭이 큰 순서로 정렬
        gap_analysis.sort(key=lambda x: x['gap'], reverse=True)
        
        # 상위 10개 출력
        for item in gap_analysis[:10]:
            gap_indicator = "⚠️" if item['gap'] > 5 else "✓"
            print(f"{gap_indicator} {item['skill']:<23} "
                  f"{item['market_demand']:>10} "
                  f"{item['current_count']:>10} "
                  f"{item['gap']:>10}")
        
        return gap_analysis
    
    def generate_training_plan(self, gap_analysis):
        """교육 계획 생성"""
        
        print("\n" + "="*70)
        print("추천 교육 계획")
        print("="*70)
        
        # 갭이 큰 스킬 Top 5
        critical_gaps = sorted(
            gap_analysis, 
            key=lambda x: x['gap'], 
            reverse=True
        )[:5]
        
        print("\n우선순위 높은 교육 과정:")
        print("-"*70)
        
        for i, item in enumerate(critical_gaps, 1):
            priority = "🔴 긴급" if item['gap'] > 10 else "🟡 중요"
            
            print(f"\n{i}. {item['skill']}")
            print(f"   우선순위: {priority}")
            print(f"   현재 보유: {item['current_count']}명")
            print(f"   필요 인원: {item['gap']}명 추가 교육 필요")
            print(f"   충족률: {item['coverage']:.1f}%")
            
            # 교육 방안 제안
            if item['gap'] > 10:
                print(f"   제안: 집체 교육 프로그램 신규 개설")
            elif item['gap'] > 5:
                print(f"   제안: 온라인 교육 과정 제공")
            else:
                print(f"   제안: 자율 학습 지원")
        
        return critical_gaps

# ══════════════════════════════════════════
#  PART 2: 실행 예제
# ══════════════════════════════════════════

def main():
    """메인 실행 함수"""

def create_skill_matrix(employees_df):
    """
    직원-스킬 매트릭스 생성
    
    Parameters:
        employees_df: DataFrame with 'name' and 'skills_list'

        DataFrame: 직원 x 스킬 매트릭스

        matrix.append(skills_dict)
    
    return pd.DataFrame(matrix).set_index('name')

def calculate_team_coverage(team_skills, required_skills):
    """
    팀 스킬 커버리지 계산
    
    Parameters:
        team_skills: set - 팀 보유 스킬
        required_skills: set - 필요 스킬
    
    Returns:
        dict: 커버리지 정보
    """
    covered = team_skills & required_skills
    missing = required_skills - team_skills
    extra = team_skills - required_skills
    
    coverage_rate = (len(covered) / len(required_skills) * 100) \
                    if required_skills else 100
    
    return {
        'coverage_rate': coverage_rate,
        'covered_skills': covered,
        'missing_skills': missing,
        'extra_skills': extra,
        'covered_count': len(covered),
        'missing_count': len(missing)
    }

    def analyze_team(self, employee_ids: List[str]) -> Dict:
        """팀 전체 역량 분석"""
        
        print("="*70)
        print("팀 역량 분석")
        print("="*70)
        
        # 팀원별 레벨 집계
        team_levels = []
        for emp_id in employee_ids:
            level = self.assessment_system.get_employee_level(emp_id)
            team_levels.append(level)
        
        # 레벨 분포
        level_distribution = Counter([
            lv.value if lv else "미인증" 
            for lv in team_levels
        ])
        
        # 역량별 커버리지 계산
        competency_coverage = self._calculate_competency_coverage(

        weak_areas = self._identify_weak_areas(competency_coverage)
        
        return {
            'team_size': len(employee_ids),
            'level_distribution': dict(level_distribution),
            'competency_coverage': competency_coverage,
            'weak_areas': weak_areas,
            'average_coverage': np.mean(list(competency_coverage.values()))
        }
    
    def _calculate_competency_coverage(self, 
                                      employee_ids: List[str]) -> Dict:
        """역량별 커버리지 계산"""
        
        coverage = {}
        
        for comp in self.ontology.competencies:
            if not isinstance(comp, DomainAXCompetency):
                continue
            
            comp_id = comp.identifier
            qualified_count = 0
            
            for emp_id in employee_ids:
                assessments = self.assessment_system.assessments.get(emp_id, [])
                
                for assessment in assessments:
                    if (assessment.competency_id == comp_id and 
                        assessment.score >= 70):
                        qualified_count += 1
                        break
            
            coverage[comp.name] = (
                qualified_count / len(employee_ids) * 100
            ) if employee_ids else 0
        
        return coverage
    
    def _identify_weak_areas(self, 
                           competency_coverage: Dict,
                           threshold: float = 50.0) -> List[str]:
        """취약 영역 식별"""
        
        weak = []
        
        for comp_name, coverage in competency_coverage.items():
            if coverage < threshold:
                weak.append({
                    'competency': comp_name,
                    'coverage': coverage,
                    'gap': threshold - coverage
                })
        
        # 갭이 큰 순으로 정렬
        weak.sort(key=lambda x: x['gap'], reverse=True)
        
        return weak
    
    def recommend_team_training(self, 
                               team_analysis: Dict,
                               budget_hours: int = 80) -> List[Dict]:
        """팀 교육 계획 추천"""
        
        print("\n" + "="*70)
        print("팀 교육 계획 추천")
        print("="*70)
        
        weak_areas = team_analysis['weak_areas']
        
        # 우선순위별 교육 과정 구성
        training_plan = []
        total_hours = 0
        
        for i, weak in enumerate(weak_areas):
            if total_hours >= budget_hours:
                break
            
            # 역량 찾기
            comp = None
            for c in self.ontology.competencies:
                if isinstance(c, DomainAXCompetency) and c.name == weak['competency']:
                    comp = c
                    break
            
            if not comp:
                continue
            
            # 교육 시간 추정 (레벨 및 갭에 따라)
            base_hours = 8
            if comp.level == AXLevel.LEVEL_2:
                base_hours = 16
            elif comp.level == AXLevel.LEVEL_3:
                base_hours = 24
            
            # 갭에 따른 조정
            adjusted_hours = int(base_hours * (weak['gap'] / 50))
            
            if total_hours + adjusted_hours <= budget_hours:
                training_plan.append({
                    'priority': i + 1,
                    'competency': weak['competency'],
                    'level': comp.level.value if comp.level else "",
                    'current_coverage': weak['coverage'],
                    'target_coverage': 80.0,
                    'estimated_hours': adjusted_hours,
                    'category': comp.category_middle,
                    'measurement_items': comp.measurement_items[:3]
                })
                
                total_hours += adjusted_hours
        
        return training_plan

class SystemConnector(ABC):

        self.config = config

        """시스템 연결"""

        """교육 이력 가져오기"""

        self.api_url = config.get('api_url')

    def connect(self) -> bool:
        """HRM 시스템 연결"""
        
        print("HRM 시스템 연결 중...")
        
        try:
            self.session = requests.Session()
            self.session.headers.update({
                'Authorization': f'Bearer {self.api_key}',
                'Content-Type': 'application/json'
            })
            
            # 연결 테스트
            response = self.session.get(f"{self.api_url}/health")
            
            if response.status_code == 200:
                print("✓ HRM 시스템 연결 성공")
                return True
            else:
                print(f"✗ HRM 연결 실패: {response.status_code}")
                return False
                
        except Exception as e:
            print(f"✗ HRM 연결 오류: {e}")
            return False

        """직원 데이터 가져오기"""
        
        if not self.session:

            response = self.session.get(f"{self.api_url}/employees")

            employees = response.json()

class DomainAXSyncManager:
    """Domain AX 데이터 동기화 관리자"""
    
    def __init__(self, 
                 ontology: DomainAXOntology,
                 assessment_system: DomainAXAssessmentSystem):
        self.ontology = ontology
        self.assessment_system = assessment_system
        self.connectors = {}
        self.sync_history = []
    
    def register_connector(self, name: str, connector: SystemConnector):
        """시스템 커넥터 등록"""
        
        self.connectors[name] = connector
        print(f"✓ {name} 커넥터 등록 완료")
    
    def sync_from_hrm(self):
        """HRM 시스템에서 데이터 동기화"""
        
        if 'HRM' not in self.connectors:
            print("✗ HRM 커넥터가 등록되지 않았습니다.")
            return
        
        print("="*70)
        print("HRM 시스템 동기화 시작")
        print("="*70)
        
        hrm = self.connectors['HRM']
        
        # 연결
        if not hrm.connect():
            return
        
        # 직원 데이터 수집
        employees = hrm.fetch_employee_data()
        
        # 각 직원의 교육 이력 수집
        for employee in employees:
            emp_id = employee['employee_id']
            
            trainings = hrm.fetch_training_records(emp_id)
            
            print(f"\n직원: {employee['name']} ({emp_id})")
            print(f"  교육 이력: {len(trainings)}건")
        
        # 동기화 기록
        self.sync_history.append({
            'timestamp': datetime.now().isoformat(),
            'source': 'HRM',
            'employees_synced': len(employees),
            'status': 'success'
        })
        
        print("\n✓ HRM 동기화 완료")
    
    def sync_from_lms(self):
        """LMS 시스템에서 데이터 동기화 및 역량 자동 평가"""
        
        if 'LMS' not in self.connectors:
            print("✗ LMS 커넥터가 등록되지 않았습니다.")
            return
        
        print("="*70)
        print("LMS 시스템 동기화 시작")
        print("="*70)
        
        lms = self.connectors['LMS']
        
        # 연결
        if not lms.connect():
            return
        
        # 학습자 데이터 수집
        learners = lms.fetch_employee_data()
        
        updated_count = 0
        
        # 각 학습자의 교육 이력 분석
        for learner in learners:
            emp_id = learner['employee_id']
            
            trainings = lms.fetch_training_records(emp_id)
            
            # 완료된 과정만 필터링
            completed = [t for t in trainings if t['status'] == 'completed']
            
            print(f"\n학습자: {learner['name']} ({emp_id})")
            print(f"  완료 과정: {len(completed)}건")
            
            # 교육 과정을 역량 평가로 변환
            new_assessments = []
            
            for training in completed:
                # 과정 ID를 역량 ID로 매핑
                comp_id = lms.map_course_to_competency(training['course_id'])
                
                if comp_id:
                    # 역량 찾기
                    comp = None
                    for c in self.ontology.competencies:
                        if isinstance(c, DomainAXCompetency) and c.identifier == comp_id:
                            comp = c
                            break
                    
                    if comp:
                        # 평가 객체 생성
                        assessment = CompetencyAssessment(
                            competency_id=comp_id,
                            competency_name=comp.name,
                            level=comp.level,
                            score=training['final_score'] if training['final_score'] else 80.0,
                            assessment_date=training['completion_date'],
                            evidence=[f"LMS 과정: {training['course_name']}"]
                        )
                        
                        new_assessments.append(assessment)
            
            # 평가 데이터 업데이트
            if new_assessments:
                existing = self.assessment_system.assessments.get(emp_id, [])
                
                # 중복 제거 (같은 역량은 최신 것만)
                comp_ids = set()
                merged = []
                
                for assessment in new_assessments + existing:
                    if assessment.competency_id not in comp_ids:
                        merged.append(assessment)
                        comp_ids.add(assessment.competency_id)
                
                self.assessment_system.assessments[emp_id] = merged
                updated_count += 1
                
                print(f"  → {len(new_assessments)}개 역량 평가 추가")
        
        # 동기화 기록
        self.sync_history.append({
            'timestamp': datetime.now().isoformat(),
            'source': 'LMS',
            'learners_synced': len(learners),
            'assessments_updated': updated_count,
            'status': 'success'
        })
        
        print(f"\n✓ LMS 동기화 완료 ({updated_count}명 역량 업데이트)")
    
    def sync_to_hrm(self):
        """Domain AX 역량 데이터를 HRM에 동기화"""
        
        if 'HRM' not in self.connectors:
            print("✗ HRM 커넥터가 등록되지 않았습니다.")
            return
        
        print("="*70)
        print("HRM으로 역량 데이터 동기화")
        print("="*70)
        
        hrm = self.connectors['HRM']
        
        if not hrm.connect():
            return
        
        success_count = 0
        
        for emp_id, assessments in self.assessment_system.assessments.items():
            # 현재 레벨 계산
            current_level = self.assessment_system.get_employee_level(emp_id)
            
            # HRM으로 전송할 데이터 구성
            competency_data = {
                'domain_ax_level': current_level.value if current_level else "미인증",
                'assessed_competencies': len(assessments),
                'last_assessment_date': max(
                    [a.assessment_date for a in assessments]
                ) if assessments else None,
                'competencies': [
                    {
                        'id': a.competency_id,
                        'name': a.competency_name,
                        'level': a.level.value,
                        'score': a.score
                    }
                    for a in assessments
                ]
            }
            
            # HRM 업데이트
            if hrm.update_competency_data(emp_id, competency_data):
                success_count += 1
        
        print(f"\n✓ {success_count}명의 역량 데이터 HRM 동기화 완료")
    
    def print_sync_history(self):
        """동기화 이력 출력"""
        
        print("="*70)
        print("동기화 이력")
        print("="*70)
        
        if not self.sync_history:
            print("동기화 이력이 없습니다.")
            return
        
        for i, record in enumerate(self.sync_history, 1):
            print(f"\n[{i}] {record['timestamp']}")
            print(f"  소스: {record['source']}")
            print(f"  상태: {record['status']}")
            
            for key, value in record.items():
                if key not in ['timestamp', 'source', 'status']:
                    print(f"  {key}: {value}")

                'month': month,
                'competencies': [comp.name for comp in month_comps],
                'total_hours': sum(
                    self._estimate_learning_hours(comp)

    def load_ontology_to_db(self, ontology: DomainAXOntology):
        """온톨로지 데이터를 데이터베이스로 로드"""
        
        session = self.get_session()
        
        try:
            print("="*70)

            print("="*70)
            
            # 기존 데이터 삭제 (개발 환경에서만)
            # session.query(Competency).delete()
            # session.query(KnowledgeItem).delete()
            # session.query(SkillItem).delete()

            skill_cache = {}
            
            competency_count = 0
            
            for comp in ontology.competencies:
                if not isinstance(comp, DomainAXCompetency):
                    continue
                
                # 역량 생성
                db_comp = Competency(
                    identifier=comp.identifier,
                    name=comp.name,
                    description=comp.description,
                    category_major=comp.category_major,
                    category_middle=comp.category_middle,
                    category_minor=comp.category_minor,
                    level=LevelEnum(comp.level.value) if comp.level else None
                )
                
                session.add(db_comp)
                session.flush()  # ID 생성
                
                # 측정 항목 추가
                for i, item in enumerate(comp.measurement_items):
                    measurement = MeasurementItem(
                        competency_id=db_comp.id,
                        description=item,
                        order=i + 1
                    )
                    session.add(measurement)
                
                # 지식 항목 추가
                for knowledge_name in comp.required_knowledge:
                    if knowledge_name not in knowledge_cache:
                        knowledge = KnowledgeItem(name=knowledge_name)
                        session.add(knowledge)
                        session.flush()
                        knowledge_cache[knowledge_name] = knowledge
                    
                    db_comp.knowledge_items.append(knowledge_cache[knowledge_name])
                
                # 기술 항목 추가
                for skill_name in comp.required_skills:
                    if skill_name not in skill_cache:
                        skill = SkillItem(name=skill_name)
                        session.add(skill)
                        session.flush()
                        skill_cache[skill_name] = skill
                    
                    db_comp.skill_items.append(skill_cache[skill_name])
                
                competency_count += 1
                
                if competency_count % 10 == 0:
                    print(f"  진행: {competency_count}개 역량 로드...")
            
            session.commit()
            
            print(f"\n✓ 총 {competency_count}개 역량 로드 완료")
            print(f"✓ {len(knowledge_cache)}개 지식 항목")
            print(f"✓ {len(skill_cache)}개 기술 항목")
            
        except Exception as e:
            session.rollback()
            print(f"✗ 데이터 로딩 오류: {e}")
            raise
        
        finally:
            session.close()

    def get_employee_stats(self, employee_id: str) -> Dict:
        """직원 통계 조회"""
        
        session = self.get_session()
        
        try:
            employee = session.query(Employee).filter_by(employee_id=employee_id).first()
            
            if not employee:
                return None
            
            assessments = session.query(Assessment).filter_by(employee_id=employee.id).all()
            
            total = len(assessments)
            passed = len([a for a in assessments if a.score >= 70])
            avg_score = sum(a.score for a in assessments) / total if total > 0 else 0
            
            return {
                'employee_id': employee_id,
                'name': employee.name,
                'total_assessments': total,
                'passed_assessments': passed,
                'average_score': round(avg_score, 1)
            }
            
        finally:
            session.close()

                employee_id=employee.id,
                target_level=target_level,
                start_date=datetime.utcnow(),
                status='planned'
            )
            
            session.add(learning_path)
            session.flush()
            
            # 학습 항목 추가
            for i, comp_id in enumerate(competency_ids):
                competency = session.query(Competency).filter_by(identifier=comp_id).first()
                
                if competency:
                    item = LearningPathItem(
                        learning_path_id=learning_path.id,

                        order=i + 1,
                        estimated_hours=8 if competency.level == LevelEnum.LEVEL_1

                                       else 24,
                        status='pending'
                    )
                    session.add(item)
            
            session.commit()
            
            print(f"✓ 학습 경로 생성 완료: ID {learning_path.id}")
            
            return learning_path.id
            
        except Exception as e:
            session.rollback()
            print(f"✗ 학습 경로 생성 오류: {e}")
            raise
        
        finally:
            session.close()

                 ontology: DomainAXOntology,
                 assessment_system: DomainAXAssessmentSystem):
        self.ontology = ontology
        self.assessment_system = assessment_system
        
        # 스타일 설정
        plt.style.use('seaborn-v0_8-darkgrid')
        sns.set_palette("husl")
    
    def generate_competency_matrix(self) -> pd.DataFrame:
        """직원×역량 매트릭스 생성"""
        
        # 모든 역량 ID
        comp_ids = [
            c.identifier 
            for c in self.ontology.competencies 
            if isinstance(c, DomainAXCompetency)
        ]
        
        # 모든 직원 ID
        emp_ids = list(self.assessment_system.assessments.keys())
        
        # 매트릭스 초기화
        matrix = pd.DataFrame(
            index=emp_ids,
            columns=comp_ids,
            data=0.0
        )
        
        # 평가 점수로 채우기
        for emp_id, assessments in self.assessment_system.assessments.items():
            for assessment in assessments:
                matrix.loc[emp_id, assessment.competency_id] = assessment.score
        
        return matrix
    
    def calculate_skill_coverage(self) -> pd.DataFrame:
        """역량별 커버리지 계산"""
        
        matrix = self.generate_competency_matrix()
        
        coverage_data = []
        
        for comp in self.ontology.competencies:
            if not isinstance(comp, DomainAXCompetency):
                continue
            
            comp_id = comp.identifier
            
            if comp_id in matrix.columns:
                scores = matrix[comp_id]
                
                total_employees = len(scores)
                qualified = (scores >= 70).sum()
                coverage = (qualified / total_employees * 100) if total_employees > 0 else 0
                avg_score = scores[scores > 0].mean() if (scores > 0).any() else 0
                
                coverage_data.append({
                    '역량ID': comp_id,
                    '역량명': comp.name,
                    '레벨': comp.level.value if comp.level else "",
                    '카테고리': comp.category_middle,
                    '보유인원': int(qualified),
                    '전체인원': total_employees,
                    '커버리지(%)': round(coverage, 1),
                    '평균점수': round(avg_score, 1)
                })
        
        df = pd.DataFrame(coverage_data)
        
        return df.sort_values('커버리지(%)', ascending=False)
    
    def analyze_competency_trends(self, 
                                  months: int = 6) -> Dict[str, pd.DataFrame]:
        """역량 트렌드 분석 (시계열)"""
        
        # 실제로는 assessment_date를 활용한 시계열 분석
        # 여기서는 시뮬레이션
        
        results = {}
        
        # 월별 평가 건수 추이
        monthly_counts = []
        
        for i in range(months):
            month_date = datetime.now() - timedelta(days=30 * i)
            count = np.random.randint(10, 50)  # 시뮬레이션
            
            monthly_counts.append({
                '월': month_date.strftime('%Y-%m'),
                '평가건수': count
            })
        
        results['monthly_counts'] = pd.DataFrame(monthly_counts)
        
        # 레벨별 인증 추이
        level_trends = []
        
        for i in range(months):
            month_date = datetime.now() - timedelta(days=30 * i)
            
            level_trends.append({
                '월': month_date.strftime('%Y-%m'),
                'Lv.1': np.random.randint(5, 15),
                'Lv.2': np.random.randint(3, 10),
                'Lv.3': np.random.randint(1, 5)
            })
        
        results['level_trends'] = pd.DataFrame(level_trends)
        
        return results
    
    def identify_skill_gaps(self, 
                          threshold: float = 50.0) -> pd.DataFrame:
        """스킬 갭 식별"""
        
        coverage_df = self.calculate_skill_coverage()
        
        # 커버리지가 임계값 미만인 역량
        gaps = coverage_df[coverage_df['커버리지(%)'] < threshold].copy()
        
        # 우선순위 계산 (커버리지 낮고, 레벨 높을수록 우선순위 높음)
        level_priority = {'Lv.3': 3, 'Lv.2': 2, 'Lv.1': 1, '': 0}
        gaps['레벨우선순위'] = gaps['레벨'].map(level_priority)
        gaps['갭점수'] = (threshold - gaps['커버리지(%)']) * gaps['레벨우선순위']
        
        return gaps.sort_values('갭점수', ascending=False)
    
    def benchmark_employees(self) -> pd.DataFrame:
        """직원 벤치마킹"""
        
        matrix = self.generate_competency_matrix()
        
        benchmark_data = []
        
        for emp_id in matrix.index:
            scores = matrix.loc[emp_id]
            
            total_assessed = (scores > 0).sum()
            passed = (scores >= 70).sum()
            avg_score = scores[scores > 0].mean() if (scores > 0).any() else 0
            
            # 레벨별 집계
            lv1_count = 0
            lv2_count = 0
            lv3_count = 0
            
            for comp in self.ontology.competencies:
                if not isinstance(comp, DomainAXCompetency):
                    continue
                
                if comp.identifier in scores.index:
                    score = scores[comp.identifier]
                    
                    if score >= 70:
                        if comp.level == AXLevel.LEVEL_1:
                            lv1_count += 1
                        elif comp.level == AXLevel.LEVEL_2:
                            lv2_count += 1
                        elif comp.level == AXLevel.LEVEL_3:
                            lv3_count += 1
            
            # 현재 레벨
            current_level = self.assessment_system.get_employee_level(emp_id)
            
            benchmark_data.append({
                '직원ID': emp_id,
                '현재레벨': current_level.value if current_level else "미인증",
                '총평가수': total_assessed,
                '통과수': passed,
                '통과율(%)': round(passed / total_assessed * 100, 1) if total_assessed > 0 else 0,
                '평균점수': round(avg_score, 1),
                'Lv.1': lv1_count,
                'Lv.2': lv2_count,
                'Lv.3': lv3_count,
                '총역량': lv1_count + lv2_count + lv3_count
            })
        
        df = pd.DataFrame(benchmark_data)
        
        return df.sort_values('총역량', ascending=False)
    
    def visualize_competency_heatmap(self, figsize=(16, 10)):
