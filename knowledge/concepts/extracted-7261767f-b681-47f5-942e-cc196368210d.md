```python
import requests

class ESCOIntegration:
    """ESCO API를 활용한 스킬 온톨로지 구축"""
    
    BASE_URL = "https://ec.europa.eu/esco/api"
    
    def fetch_skills(self, occupation_uri: str) -> List[Dict]:
        """특정 직업에 필요한 스킬 가져오기"""
        response = requests.get(
            f"{self.BASE_URL}/resource/occupation",
            params={'uri': occupation_uri, 'language': 'en'}
        )
        
        skills = []
        for skill_ref in response.json().get('hasEssentialSkill', []):
            skill_detail = self.get_skill_detail(skill_ref['uri'])
            skills.append({
                'esco_uri': skill_ref['uri'],
                'name': skill_detail['preferredLabel'],
                'description': skill_detail.get('description'),
                'skill_type': skill_detail.get('skillType'),  # knowledge/skill/competence
                'reusability_level': skill_detail.get('skillReusabilityLevel'),
                'related_skills': skill_detail.get('broaderSkill', [])
            })
        
        return skills
    
    def build_initial_graph(self):
        """초기 3,000 노드 그래프 구축"""
        
        # 1. 주요 직업군 선택 (제조, IT, HR 중심)
        target_occupations = [
            'http://data.europa.eu/esco/occupation/...',  # Software Developer
            'http://data.europa.eu/esco/occupation/...',  # Data Analyst
            'http://data.europa.eu/esco/occupation/...',  # HR Manager
            'http://data.europa.eu/esco/occupation/...',  # Production Manager
            # ... 50개 직업
        ]
        
        # 2. 각 직업의 필수/선택 스킬 추출
        all_skills = set()
        for occupation in target_occupations:
            skills = self.fetch_skills(occupation)
            all_skills.update(skills)
        
        # 3. 스킬 간 관계 매핑
        skill_graph = self.map_skill_relationships(all_skills)
        
        # 4. Firestore/Neo4j에 저장
        self.save_to_database(skill_graph)
        
        return skill_graph
```

```python
class Neo4jOntology:
    """Neo4j 기반 고급 그래프 분석"""
    
    def __init__(self):
        self.driver = GraphDatabase.driver(
            "neo4j://neo4j-instance:7687",
            auth=("neo4j", "password")
        )
    
    def find_shortest_learning_path(self, current_skills, target_skill):
        """최단 학습 경로 (Neo4j Cypher)"""
        with self.driver.session() as session:
            result = session.run("""
                MATCH (start:Skill) WHERE start.id IN $current_skills
                MATCH (end:Skill {id: $target_skill})
                MATCH path = shortestPath((start)-[:ENABLES*..10]->(end))
                RETURN path, length(path) as steps
                ORDER BY steps ASC
                LIMIT 1
            """, current_skills=current_skills, target_skill=target_skill)
            
            return result.single()['path']
    
    def recommend_skills(self, employee_id, role_target):
        """역할 전환을 위한 스킬 추천 (PageRank 기반)"""
        with self.driver.session() as session:
            result = session.run("""
                // 1. 목표 역할에 필요한 스킬들
                MATCH (role:Role {id: $role_target})-[:REQUIRES]->(skill:Skill)
                
                // 2. 직원이 이미 가진 스킬들
                MATCH (emp:Employee {id: $employee_id})-[:HAS_SKILL]->(current:Skill)
                
                // 3. 없는 스킬 중 현재 스킬과 가장 관련 높은 것
                // (PageRank로 중요도 계산)
                CALL gds.pageRank.stream({
                    nodeProjection: 'Skill',
                    relationshipProjection: {
                        ENABLES: {type: 'ENABLES', orientation: 'NATURAL'}
                    }
                })
                YIELD nodeId, score
                WITH gds.util.asNode(nodeId) AS skill, score
                WHERE skill IN (skill - current)
                RETURN skill, score
                ORDER BY score DESC
                LIMIT 10
            """, employee_id=employee_id, role_target=role_target)
            
            return [r['skill'] for r in result]
```

```python
class ESCOClient:
    """ESCO API 클라이언트"""
    
    def __init__(self):
        self.base_url = settings.esco_base_url
        self.language = settings.esco_language
        self.session = requests.Session()
        
    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=4, max=10)
    )
    def _make_request(self, endpoint: str, params: Dict) -> Dict:
        """API 요청 with 재시도"""
        url = f"{self.base_url}/{endpoint}"
        params['language'] = self.language
        
        try:
            response = self.session.get(url, params=params, timeout=30)
            response.raise_for_status()
            
            # 캐시 여부 로깅
            cache_status = "HIT" if getattr(response, 'from_cache', False) else "MISS"
            logger.debug(f"API {endpoint} [Cache: {cache_status}]")
            
            return response.json()
            
        except requests.exceptions.RequestException as e:
            logger.error(f"ESCO API error: {e}")
            raise
    
    def search_occupations(self, query: str, limit: int = 10) -> List[Dict]:
        """직업 검색"""
        data = self._make_request('suggest', {
            'text': query,
            'type': 'occupation',
            'limit': limit
        })
        
        return data.get('_embedded', {}).get('results', [])
    
    def get_occupation(self, uri: str) -> Dict:
        """직업 상세 정보"""
        return self._make_request('resource/occupation', {'uri': uri})
    
    def get_skill(self, uri: str) -> Dict:
        """스킬 상세 정보"""
        return self._make_request('resource/skill', {'uri': uri})

        """직업에 필요한 스킬 목록"""

                essential_skills.append(skill)
```

```python
@dataclass
class SkillNode:
    """스킬 노드 데이터 모델"""
    uri: str
    name: str
    description: Optional[str] = None
    skill_type: Optional[str] = None  # 'knowledge', 'skill', 'competence'
    reusability_level: Optional[str] = None  # 'cross-sector', 'sector-specific', etc.

    narrower_skills: List[str] = field(default_factory=list)
    related_skills: List[str] = field(default_factory=list)
    
    # 직업 연결
    required_by_occupations: List[Dict] = field(default_factory=list)  # {'uri': ..., 'essential': True/False}
    
    # 메타데이터
    source: str = 'ESCO'
    created_at: datetime = field(default_factory=datetime.now)
    
    def to_dict(self) -> Dict:
        """Firestore 저장용 딕셔너리"""
        return {
            'uri': self.uri,
            'name': self.name,
            'description': self.description,
            'skill_type': self.skill_type,
            'reusability_level': self.reusability_level,
            'broader_skills': self.broader_skills,
            'narrower_skills': self.narrower_skills,
            'related_skills': self.related_skills,
            'required_by_occupations': self.required_by_occupations,
            'source': self.source,
            'created_at': self.created_at.isoformat()
        }


class OntologyBuilder:
    """온톨로지 빌더"""
    
    def __init__(self):
        self.graph = nx.DiGraph()
        self.skills: Dict[str, SkillNode] = {}
        self.occupations: Dict[str, Dict] = {}
        
    def add_occupation(self, occupation_uri: str) -> None:
        """직업 추가 및 관련 스킬 수집"""
        logger.info(f"Adding occupation: {occupation_uri}")
        
        try:
            # 직업 정보
            occupation = esco_client.get_occupation(occupation_uri)
            self.occupations[occupation_uri] = {
                'uri': occupation_uri,
                'name': occupation.get('preferredLabel', {}).get(settings.esco_language, ''),
                'description': occupation.get('description', {}).get(settings.esco_language, '')
            }
            
            # 스킬 수집
            skills_data = esco_client.get_occupation_skills(occupation_uri)
            
            # Essential skills
            for skill in tqdm(skills_data['essential'], desc="Essential skills", leave=False):
                self._add_skill(skill, occupation_uri, essential=True)
            
            # Optional skills (제한)
            max_optional = settings.max_skills_per_occupation - len(skills_data['essential'])
            for skill in tqdm(skills_data['optional'][:max_optional], desc="Optional skills", leave=False):
                self._add_skill(skill, occupation_uri, essential=False)
                
        except Exception as e:
            logger.error(f"Failed to add occupation {occupation_uri}: {e}")
    
    def _add_skill(self, skill_data: Dict, occupation_uri: str, essential: bool) -> None:
        """스킬 노드 추가"""
        skill_uri = skill_data['uri']
        
        # 이미 존재하면 직업 연결만 추가
        if skill_uri in self.skills:
            self.skills[skill_uri].required_by_occupations.append({
                'uri': occupation_uri,
                'essential': essential
            })
            return
        
        # 새 스킬 노드 생성
        skill_node = SkillNode(
            uri=skill_uri,
            name=skill_data.get('preferredLabel', {}).get(settings.esco_language, ''),
            description=skill_data.get('description', {}).get(settings.esco_language),
            skill_type=skill_data.get('skillType'),
            reusability_level=skill_data.get('skillReusabilityLevel'),
            required_by_occupations=[{
                'uri': occupation_uri,
                'essential': essential
            }]
        )
        
        # 스킬 관계 수집
        try:
            relations = esco_client.get_skill_relations(skill_uri)
            skill_node.broader_skills = relations['broader']
            skill_node.narrower_skills = relations['narrower']
            skill_node.related_skills = relations['related']
        except Exception as e:
            logger.warning(f"Failed to get relations for {skill_uri}: {e}")
        
        self.skills[skill_uri] = skill_node
        
        # 그래프에 노드 추가
        self.graph.add_node(skill_uri, **skill_node.to_dict())
        
        # 엣지 추가
        for broader_uri in skill_node.broader_skills:
            self.graph.add_edge(skill_uri, broader_uri, relation='IS_A')
        
        for narrower_uri in skill_node.narrower_skills:
            self.graph.add_edge(narrower_uri, skill_uri, relation='IS_A')
        
        for related_uri in skill_node.related_skills:
            self.graph.add_edge(skill_uri, related_uri, relation='RELATED_TO')
    
    def expand_graph(self, max_depth: int = 2) -> None:
        """
        그래프 확장: 기존 스킬의 관련 스킬들을 재귀적으로 추가
        
        철학적 의미:
        - 직업에서 시작했지만, 스킬의 내재적 관계를 따라 확장
        - "연결의 연결"을 탐색하여 암묵적 지식 경로 발견
        """
        logger.info(f"Expanding graph to depth {max_depth}")
        
        current_skills = set(self.skills.keys())
        
        for depth in range(max_depth):
            new_skills = set()
            
            for skill_uri in tqdm(current_skills, desc=f"Depth {depth+1}"):
                skill_node = self.skills[skill_uri]
                
                # 관련 스킬들 추가
                related_uris = (
                    skill_node.broader_skills + 
                    skill_node.narrower_skills + 
                    skill_node.related_skills
                )
                
                for related_uri in related_uris:
                    if related_uri not in self.skills and related_uri not in new_skills:
                        try:
                            skill_data = esco_client.get_skill(related_uri)
                            self._add_skill(skill_data, occupation_uri='', essential=False)
                            new_skills.add(related_uri)
                        except Exception as e:
                            logger.warning(f"Failed to expand skill {related_uri}: {e}")
            
            if not new_skills:
                logger.info(f"No new skills at depth {depth+1}, stopping expansion")
                break
            
            current_skills = new_skills
            logger.info(f"Added {len(new_skills)} skills at depth {depth+1}")
    
    def get_statistics(self) -> Dict:
        """온톨로지 통계"""
        return {
            'total_skills': len(self.skills),
            'total_occupations': len(self.occupations),

        }

        counts = {}
        for skill in self.skills.values():
            value = getattr(skill, field, None)
            if value:
                counts[value] = counts.get(value, 0) + 1
        return counts
```

```python
from google.cloud import firestore
from google.cloud import logging as cloud_logging
from typing import Dict, List
import logging
from config import settings
from ontology_builder import OntologyBuilder
from tqdm import tqdm

logger = logging.getLogger(__name__)


class FirestoreAdapter:
    """Firestore 온톨로지 저장"""
    
    def __init__(self):
        self.db = firestore.Client(
            project=settings.gcp_project_id,
            database=settings.firestore_database
        )
        
        # Cloud Logging 설정
        logging_client = cloud_logging.Client()
        logging_client.setup_logging()
    
    def save_ontology(self, builder: OntologyBuilder) -> Dict:
        """온톨로지 저장"""
        logger.info("Saving ontology to Firestore...")
        
        stats = {
            'skills_saved': 0,
            'occupations_saved': 0,
            'edges_saved': 0,
            'errors': []
        }
        
        # 1. 메타데이터 저장
        self._save_metadata(builder)
        
        # 2. 스킬 노드 저장
        batch = self.db.batch()
        batch_count = 0
        
        for skill_uri, skill_node in tqdm(builder.skills.items(), desc="Saving skills"):
            try:
                doc_ref = self.db.collection('ontology_skills').document(
                    self._uri_to_id(skill_uri)
                )
                batch.set(doc_ref, skill_node.to_dict())
                batch_count += 1
                stats['skills_saved'] += 1
                
                # 배치 크기 제한
                if batch_count >= settings.batch_size:
                    batch.commit()
                    batch = self.db.batch()
                    batch_count = 0
                    
            except Exception as e:
                logger.error(f"Failed to save skill {skill_uri}: {e}")
                stats['errors'].append(str(e))
        
        # 남은 배치 커밋
        if batch_count > 0:
            batch.commit()
        
        # 3. 직업 저장
        for occupation_uri, occupation_data in tqdm(builder.occupations.items(), desc="Saving occupations"):
            try:
                doc_ref = self.db.collection('ontology_occupations').document(
                    self._uri_to_id(occupation_uri)
                )
                doc_ref.set(occupation_data)
                stats['occupations_saved'] += 1
            except Exception as e:
                logger.error(f"Failed to save occupation {occupation_uri}: {e}")
                stats['errors'].append(str(e))
        
        # 4. 엣지 저장 (그래프 관계)
        batch = self.db.batch()
        batch_count = 0
        
        for source, target, data in tqdm(builder.graph.edges(data=True), desc="Saving edges"):
            try:
                edge_id = f"{self._uri_to_id(source)}_{self._uri_to_id(target)}"
                doc_ref = self.db.collection('ontology_edges').document(edge_id)
                
                batch.set(doc_ref, {
                    'source': source,
                    'target': target,
                    'relation': data.get('relation', 'UNKNOWN')
                })
                batch_count += 1
                stats['edges_saved'] += 1
                
                if batch_count >= settings.batch_size:
                    batch.commit()
                    batch = self.db.batch()
                    batch_count = 0
                    
            except Exception as e:
                logger.error(f"Failed to save edge {source}->{target}: {e}")
                stats['errors'].append(str(e))
        
        if batch_count > 0:
            batch.commit()
        
        logger.info(f"Ontology saved: {stats}")
        return stats
    
    def _save_metadata(self, builder: OntologyBuilder) -> None:
        """온톨로지 메타데이터 저장"""
        metadata = {
            'version': '1.0.0',
            'source': 'ESCO',
            'statistics': builder.get_statistics(),
            'updated_at': firestore.SERVER_TIMESTAMP
        }
        
        self.db.collection('ontology_metadata').document('current').set(metadata)
    
    def _uri_to_id(self, uri: str) -> str:
        """URI를 Firestore document ID로 변환"""
        # http://data.europa.eu/esco/skill/... -> skill_...
        return uri.replace('http://data.europa.eu/esco/', '').replace('/', '_')
    
    def load_ontology(self) -> Dict:
        """저장된 온톨로지 로드"""
        logger.info("Loading ontology from Firestore...")
        
        # 메타데이터
        metadata_ref = self.db.collection('ontology_metadata').document('current')
        metadata = metadata_ref.get().to_dict()
        
        # 스킬
        skills = {}
        for doc in self.db.collection('ontology_skills').stream():
            skills[doc.id] = doc.to_dict()
        
        # 직업
        occupations = {}
        for doc in self.db.collection('ontology_occupations').stream():
            occupations[doc.id] = doc.to_dict()
        
        # 엣지
        edges = []
        for doc in self.db.collection('ontology_edges').stream():
            edges.append(doc.to_dict())
        
        return {
            'metadata': metadata,
            'skills': skills,
            'occupations': occupations,
            'edges': edges
        }
```

```python
"""
아이패드용 ESCO 온톨로지 빌더 (간소화)
"""
import requests
from google.cloud import firestore
from tqdm import tqdm
import json

# Firestore 클라이언트
db = firestore.Client()

# ESCO API 기본 설정
ESCO_BASE = "https://ec.europa.eu/esco/api"
LANG = "en"


def get_occupation(uri):
    """직업 정보 가져오기"""
    response = requests.get(
        f"{ESCO_BASE}/resource/occupation",
        params={'uri': uri, 'language': LANG},
        timeout=30
    )
    return response.json()


def get_skill(uri):
    """스킬 정보 가져오기"""
    response = requests.get(
        f"{ESCO_BASE}/resource/skill",
        params={'uri': uri, 'language': LANG},
        timeout=30
    )
    return response.json()


def collect_skills_from_occupation(occupation_uri):
    """직업에서 스킬 수집"""
    print(f"\n📊 Processing: {occupation_uri}")
    
    # 직업 정보
    occ_data = get_occupation(occupation_uri)
    occ_name = occ_data.get('preferredLabel', {}).get(LANG, 'Unknown')
    print(f"   Name: {occ_name}")
    
    # Essential 스킬
    skills = []
    essential_refs = occ_data.get('hasEssentialSkill', [])
    
    print(f"   Collecting {len(essential_refs)} essential skills...")
    
    for skill_ref in tqdm(essential_refs[:20], desc="   Skills"):  # 최대 20개로 제한
        try:
            skill_data = get_skill(skill_ref['uri'])
            
            skill_doc = {
                'uri': skill_ref['uri'],
                'name': skill_data.get('preferredLabel', {}).get(LANG, ''),
                'description': skill_data.get('description', {}).get(LANG, ''),
                'skill_type': skill_data.get('skillType'),
                'source': 'ESCO',
                'required_by': [{'occupation': occupation_uri, 'essential': True}]
            }
            
            skills.append(skill_doc)
            
        except Exception as e:
            print(f"   ⚠️  Skipped skill: {e}")
    
    return skills


def save_to_firestore(skills):
    """Firestore 저장"""
    print(f"\n💾 Saving {len(skills)} skills to Firestore...")
    
    batch = db.batch()
    count = 0
    
    for skill in skills:
        doc_id = skill['uri'].split('/')[-1]
        ref = db.collection('ontology_skills').document(doc_id)
        batch.set(ref, skill, merge=True)  # merge=True로 중복 처리
        count += 1
        
        if count % 10 == 0:
            batch.commit()
            batch = db.batch()
    
    if count % 10 != 0:
        batch.commit()
    
    print(f"✅ Saved {count} skills!")


def main():
    """메인 실행"""
    print("=" * 60)
    print("🧠 ESCO Ontology Builder for iPad")
    print("=" * 60)
    
    # 목표 직업 (5개로 시작)

        "http://data.europa.eu/esco/occupation/114e1eff-215e-47df-8e10-45a5b72f8197",  # Software Dev
        "http://data.europa.eu/esco/occupation/caf7b8fb-b2d6-46b7-bffa-bea7f8e4aed8",  # Data Analyst
        "http://data.europa.eu/esco/occupation/684fd8d5-9b40-4b02-a680-438f0082d923",  # HR Manager

    all_skills = []

        skills = collect_skills_from_occupation(occ_uri)
        all_skills.extend(skills)
    
    # 중복 제거
    unique_skills = {s['uri']: s for s in all_skills}.values()
    
    print(f"\n📈 Statistics:")
    print(f"   Total skills collected: {len(all_skills)}")
    print(f"   Unique skills: {len(unique_skills)}")
    
    # Firestore 저장
    save_to_firestore(list(unique_skills))
    
    print("\n🎉 Done! Check Firestore Console.")


if __name__ == "__main__":
```

```python
# Cell 4: ESCO 클라이언트
import requests
from google.cloud import firestore

class ESCOCollector:
    def __init__(self):
        self.db = firestore.Client()
        self.base_url = "https://ec.europa.eu/esco/api"
    
    def get_occupation_skills(self, occupation_uri):
        """직업 스킬 수집"""
        # 직업 정보
        occ = requests.get(
            f"{self.base_url}/resource/occupation",
            params={'uri': occupation_uri, 'language': 'en'}
        ).json()
        
        # 스킬 수집
        skills = []
        for skill_ref in occ.get('hasEssentialSkill', [])[:10]:
            skill = requests.get(
                f"{self.base_url}/resource/skill",
                params={'uri': skill_ref['uri'], 'language': 'en'}
            ).json()
            
            skills.append({
                'name': skill['preferredLabel']['en'],
                'uri': skill_ref['uri'],
                'type': skill.get('skillType')
            })
        
        return skills
    
    def save_skills(self, skills):
        """Firestore 저장"""
        for skill in skills:
            doc_id = skill['uri'].split('/')[-1]
            self.db.collection('ontology_skills').document(doc_id).set(skill)
        
        return len(skills)

# 인스턴스 생성
collector = ESCOCollector()
print("✅ Collector ready!")
```

```python
PROJECT_ID = "hr-ontology-mvp"

def save_skill_to_firestore(skill_data):
    """Firestore REST API로 저장"""
    url = f"https://firestore.googleapis.com/v1/projects/{PROJECT_ID}/databases/(default)/documents/ontology_skills"
    
    headers = {
        "Authorization": f"Bearer {ACCESS_TOKEN}",
        "Content-Type": "application/json"
    }
    
    # Firestore 문서 형식
    doc = {
        "fields": {
            "name": {"stringValue": skill_data['name']},
            "uri": {"stringValue": skill_data['uri']},
            "type": {"stringValue": skill_data.get('type', '')}
        }
    }
    
    response = requests.post(url, headers=headers, json=doc)
    return response.json()
```

```python
# ai_helper.py
import anthropic
import os

client = anthropic.Anthropic(
    api_key=os.environ.get("ANTHROPIC_API_KEY")
)

def generate_code(prompt: str) -> str:
    """Claude API로 코드 생성"""
    message = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=4000,
        messages=[{
            "role": "user",
            "content": f"Generate Python code for: {prompt}\n\nOnly output the code, no explanation."
        }]
    )
    
    return message.content[0].text

# 사용 예시
if __name__ == "__main__":
    code = generate_code(
        "ESCO API에서 occupation의 skills를 가져오는 함수"
    )
    
    # 파일로 저장
    with open("generated_code.py", "w") as f:
        f.write(code)
    
    print("✅ Code generated: generated_code.py")
```

```bash
cat > main.py << 'EOF'
"""ESCO → Firestore"""
import requests
from google.cloud import firestore

def main():
    db = firestore.Client()
    
    # ESCO API
    r = requests.get(
        "https://ec.europa.eu/esco/api/suggest",
        params={'text': 'python', 'type': 'skill', 'language': 'en'}
    )
    
    skills = r.json()['_embedded']['results'][:5]
    
    # Firestore 저장
    for skill in skills:
        doc_id = skill['uri'].split('/')[-1]
        db.collection('test_skills').document(doc_id).set({
            'name': skill['title'],
            'uri': skill['uri']
        })
    
    print(f"✅ Saved {len(skills)} skills to Firestore")

if __name__ == "__main__":
    main()
EOF

# 실행
python main.py

# 출력:
# ✅ Saved 5 skills to Firestore
```
