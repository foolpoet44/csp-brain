from typing import List, Optional
from sqlalchemy.orm import Session
from app.models.employee import Employee
from app.schemas.employee import EmployeeCreate, EmployeeUpdate

class EmployeeService:
    def get_employee(self, db: Session, employee_id: int) -> Optional[Employee]:
        return db.query(Employee).filter(Employee.id == employee_id).first()
    
    def get_employees(
        self, 
        db: Session, 
        company_id: int,
        skip: int = 0, 
        limit: int = 100,
        department: Optional[str] = None
    ) -> List[Employee]:
        query = db.query(Employee).filter(Employee.company_id == company_id)
        if department:
            query = query.filter(Employee.department == department)
        return query.offset(skip).limit(limit).all()
    
    def create_employee(self, db: Session, employee: EmployeeCreate, company_id: int) -> Employee:
        db_employee = Employee(**employee.dict(), company_id=company_id)
        db.add(db_employee)
        db.commit()
        db.refresh(db_employee)
        return db_employee
    
    def update_employee(
        self, 
        db: Session, 
        employee_id: int, 
        employee_update: EmployeeUpdate
    ) -> Optional[Employee]:
        db_employee = self.get_employee(db, employee_id)
        if db_employee:
            update_data = employee_update.dict(exclude_unset=True)
            for field, value in update_data.items():
                setattr(db_employee, field, value)
            db.commit()
            db.refresh(db_employee)
        return db_employee
    
    def delete_employee(self, db: Session, employee_id: int) -> bool:
        db_employee = self.get_employee(db, employee_id)
        if db_employee:
            db.delete(db_employee)
            db.commit()
            return True
        return False

employee_service = EmployeeService()

from datetime import date, datetime, time
from typing import List, Optional
from sqlalchemy.orm import Session
from app.models.attendance import AttendanceRecord
from app.schemas.attendance import AttendanceCreate

class AttendanceService:
    def check_in(self, db: Session, employee_id: int) -> AttendanceRecord:
        # 오늘 이미 출근 기록이 있는지 확인
        today = date.today()
        existing_record = db.query(AttendanceRecord).filter(
            AttendanceRecord.employee_id == employee_id,
            AttendanceRecord.date == today
        ).first()
        
        if existing_record and existing_record.check_in:
            raise ValueError("이미 출근 처리되었습니다")
        
        if existing_record:
            # 기존 레코드 업데이트
            existing_record.check_in = datetime.now()
            existing_record.status = self._calculate_status(existing_record.check_in)
        else:
            # 새 레코드 생성
            check_in_time = datetime.now()
            existing_record = AttendanceRecord(
                employee_id=employee_id,
                date=today,
                check_in=check_in_time,
                status=self._calculate_status(check_in_time)
            )
            db.add(existing_record)
        
        db.commit()
        db.refresh(existing_record)
        return existing_record
    
    def check_out(self, db: Session, employee_id: int) -> AttendanceRecord:
        today = date.today()
        record = db.query(AttendanceRecord).filter(
            AttendanceRecord.employee_id == employee_id,
            AttendanceRecord.date == today
        ).first()
        
        if not record or not record.check_in:
            raise ValueError("출근 기록이 없습니다")
        
        if record.check_out:
            raise ValueError("이미 퇴근 처리되었습니다")
        
        record.check_out = datetime.now()
        record.total_hours = self._calculate_total_hours(record.check_in, record.check_out)
        
        db.commit()
        db.refresh(record)
        return record
    
    def _calculate_status(self, check_in_time: datetime) -> str:
        # 9시 이후 출근은 지각 처리
        standard_time = time(9, 0)
        if check_in_time.time() > standard_time:
            return "late"
        return "present"
    
    def _calculate_total_hours(self, check_in: datetime, check_out: datetime) -> float:
        total_minutes = (check_out - check_in).total_seconds() / 60
        # 점심시간 1시간 제외
        total_hours = (total_minutes - 60) / 60
        return max(0, total_hours)

    def calculate_monthly_salary(self, db: Session, employee_id: int, year: int, month: int) -> Dict[str, Any]:
        employee = db.query(Employee).filter(Employee.id == employee_id).first()
        if not employee:
            raise ValueError("직원을 찾을 수 없습니다")
        
        # 해당 월의 출근 기록 조회
        attendance_records = db.query(AttendanceRecord).filter(
            AttendanceRecord.employee_id == employee_id,
            AttendanceRecord.date >= date(year, month, 1),
            AttendanceRecord.date < date(year, month + 1, 1) if month < 12 else date(year + 1, 1, 1)
        ).all()
        
        # 기본 계산
        base_salary = employee.salary
        total_work_days = len([r for r in attendance_records if r.status == "present" or r.status == "late"])
        total_hours = sum(r.total_hours or 0 for r in attendance_records)
        
        # 초과근무 계산 (월 160시간 기준)
        standard_hours = 160
        overtime_hours = max(0, total_hours - standard_hours)
        overtime_pay = int(overtime_hours * (base_salary / standard_hours) * 1.5)
        
        # 공제액 계산 (간단한 예시)
        national_pension = int(base_salary * 0.045)  # 국민연금 4.5%
        health_insurance = int(base_salary * 0.03335)  # 건강보험 3.335%
        employment_insurance = int(base_salary * 0.008)  # 고용보험 0.8%
        income_tax = int((base_salary + overtime_pay) * 0.08)  # 소득세 8% (간소화)
        
        total_deductions = national_pension + health_insurance + employment_insurance + income_tax
        net_salary = base_salary + overtime_pay - total_deductions
        
        return {
            "employee_id": employee_id,
            "year_month": f"{year}-{month:02d}",
            "base_salary": base_salary,
            "overtime_hours": overtime_hours,
            "overtime_pay": overtime_pay,
            "total_work_days": total_work_days,
            "deductions": {
                "national_pension": national_pension,
                "health_insurance": health_insurance,
                "employment_insurance": employment_insurance,
                "income_tax": income_tax,
                "total": total_deductions
            },
            "net_salary": net_salary
        }

            overtime_pay=calculation_result["overtime_pay"],
            deductions=calculation_result["deductions"]["total"],
            net_salary=calculation_result["net_salary"],
            status="confirmed"
        )
        
        db.add(payroll_record)
        db.commit()
        db.refresh(payroll_record)

    department: '',
    status: '',
    search: ''
  });

    fetchEmployees();
  }, [filters]);

  const fetchEmployees = async () => {
    try {
      setLoading(true);
      const response = await employeeService.getEmployees(filters);
      setEmployees(response.data);
    } catch (error) {
      console.error('직원 목록 조회 실패:', error);

      setLoading(false);

    def format(self, record):
        log_obj = {
            'timestamp': datetime.utcnow().isoformat(),
            'level': record.levelname,
            'message': record.getMessage(),
            'module': record.module,
            'function': record.funcName,
            'line': record.lineno

            log_obj['user_id'] = record.user_id

            log_obj['request_id'] = record.request_id

from typing import List, Optional
from fastapi import Depends, Query
from sqlalchemy.orm import Session, selectinload
from app.models.employee import Employee

@router.get("/employees/optimized", response_model=List[EmployeeOptimized])
def get_employees_optimized(
    page: int = Query(1, ge=1),
    size: int = Query(20, ge=1, le=100),
    department: Optional[str] = None,
    db: Session = Depends(get_db)
):
    """
    성능 최적화된 직원 목록 조회
    - 페이지네이션 적용
    - 필요한 필드만 선택
    - 즉시 로딩으로 N+1 문제 해결
    """
    query = db.query(Employee).options(
        selectinload(Employee.department),  # 관련 데이터 즉시 로딩
    )
    
    if department:
        query = query.filter(Employee.department == department)
    
    # 페이지네이션
    offset = (page - 1) * size
    employees = query.offset(offset).limit(size).all()
    
    return employees

from fastapi import HTTPException, Request
import time
from collections import defaultdict

class SecurityMiddleware:
    def __init__(self):
        self.rate_limit_storage = defaultdict(list)
    
    async def rate_limit_check(self, request: Request):
        """API 호출 제한 (Rate Limiting)"""
        client_ip = request.client.host
        current_time = time.time()
        
        # 최근 1분간의 요청만 유지
        self.rate_limit_storage[client_ip] = [
            req_time for req_time in self.rate_limit_storage[client_ip]
            if current_time - req_time < 60
        ]
        
        # 분당 100회 제한
        if len(self.rate_limit_storage[client_ip]) >= 100:
            raise HTTPException(
                status_code=429, 
                detail="Too many requests. Please try again later."
            )
        
        self.rate_limit_storage[client_ip].append(current_time)
    
    def validate_input(self, data: dict):
        """입력값 검증"""
        for key, value in data.items():
            if isinstance(value, str):
                # SQL Injection 패턴 검사
                dangerous_patterns = [
                    "DROP TABLE", "DELETE FROM", "UPDATE SET",
                    "INSERT INTO", "UNION SELECT", "--", "/*", "*/"
                ]
                value_upper = value.upper()
                for pattern in dangerous_patterns:
                    if pattern in value_upper:
                        raise HTTPException(
                            status_code=400,
                            detail=f"Invalid input detected: {key}"
                        )

    )
    
    def calculate_total_hours(self):

        if self.check_in and self.check_out:
            total_minutes = (self.check_out - self.check_in).total_seconds() / 60
            work_minutes = total_minutes - (self.break_minutes or 0)
            self.total_hours = max(0, work_minutes / 60)

                self.overtime_hours = self.total_hours - 8
        return self.total_hours

        """ID로 직원 조회"""

        self, 
        db: Session, 
        department_id: Optional[int] = None,
        status: Optional[EmployeeStatus] = None,
        search: Optional[str] = None,
        skip: int = 0,
        limit: int = 100
    ) -> List[Employee]:
        """조건별 직원 목록 조회"""
        
        query = db.query(Employee)
        
        # 조건 추가
        if department_id:
            query = query.filter(Employee.department_id == department_id)
        
        if status:
            query = query.filter(Employee.status == status)
        
        if search:
            # 이름 또는 이메일로 검색
            query = query.filter(
                or_(
                    Employee.name.ilike(f"%{search}%"),
                    Employee.email.ilike(f"%{search}%")
                )
            )
        
        # 정렬 및 페이지네이션
        return query.order_by(Employee.created_at.desc())\
                   .offset(skip)\
                   .limit(limit)\

        self,
